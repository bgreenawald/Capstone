 نازلة العصر الخميس رمضان ه يونيو م إبراهيم المطرودي شيخ معروف وداعية مشهور وقاض قديم يلتقطون الصور، ويبعثون بالمقاطع، ويمضون في حياتهم كأن لم يأتوا منكرا، ولم يقعوا في محظور تلك صورة تبعث على التفكير في التفاوت الكبير بين ما يقوله النص ويحكى عن العلماء فيه وبين ما يسير عليه الشيخ والداعية والقاضي في حياتهم؛ هناك صدام وفير بين ما يقوم به الموجه، وهو هنا عادة الشيخ والداعية والقاضي، وبين سلوكه الذي يراه الناس منه، ويجدونه عليه، كان ينصح الناس بالتزام النص، والوقوف عنده، والصبر على الأخذ به، وهو الآن يتساهل فيه، ويتلمس أوهى الحجج لتركه والتخلي عنه، وهو تفاوت يبعث كثيرا من القضايا، ويطرح كثيرا من الأسئلة، وربما كان له دور كبير في تجديد الخطاب الديني، وإجالة النظر فيه؛ لو نظرنا فيه نظرة تأريخية، تستقصي نماذجه، وتدرس أشكاله، وتستوعب ما استطاعت مراحله، وتفتش عنه عند المذاهب والطوائف والنحل في تأريخ الإسلام التصوير نازلة من نوازل العصر، لا لأن عامة المسلمين فقط أصبحوا لا يستغنون عنه، ولا تمضي حياتهم دونه، بل هو نازلة لأن كثيرا من الشيوخ والدعاة والقضاة لا يتورعون عنه، ولا يحتجبون منه، ويسجلون دقائق حياتهم به لماذا أصبح الشيخ والداعية اللذان عهدناهما يستسلمان للنص، ويأخذان به، ويقفان عنده، ويدعوان الناس إليه؛ يخرجان عن دلالته، ويبعدان عنها، وربما يسعيان إلى تأويله، وحمله على غير ظاهره؟ لماذا يحق لشيخ وداعية أن يعفيا نفسيهما من دلالة النص، ووظيفتهما قائمة على إقناع المسلمين به، أو يؤولا نصا، ويصرفاه عن ظاهره، ويخرجا به عن دلالة من نقله إليهما، ولا يحق لغيرهما أن يقوما بمثل ما قاما به في نص آخر ومتن مختلف؟التصوير نازلة من نوازل العصر، لا لأن عامة المسلمين فقط أصبحوا لا يستغنون عنه، ولا تمضي حياتهم دونه، بل هو نازلة لأن كثيرا من الشيوخ والدعاة والقضاة لا يتورعون عنه، ولا يحتجبون منه، ويسجلون دقائق حياتهم به، ويوثقون كثيرا من أعمالهم عبره، فقد اجتمع على الأخذ به، واللجوء إليه، من ينتظر منهم أن يعظوا في تهالك الناس عليه، وانهماكهم به، ويسعوا إلى تحذيرهم منه، ومطالبتهم بتقليل الانجرار إليه يلخص النووي رحمه الله تعالى موقف العلماء والفقهاء في القديم قائلا قال أصحابنا وغيرهم من العلماء تصوير صورة الحيوان حرام شديد التحريم، وهو من الكبائر؛ لأنه متوعد عليه بهذا الوعيد الشديد المذكور في الأحاديث، وسواء صنعه بما يمتهن أو بغيره فصنعته حرام بكل حال؛ لأن فيه مضاهاة لخلق الله تعالى ولا فرق في هذا كله بين ما له ظل وما لا ظل له، هذا تلخيص مذهبنا في المسألة، وبمعناه قال جماهير العلماء من الصحابة والتابعين ومن بعدهم، وهو مذهب الثوري ومالك وأبي حنيفة وغيرهم صحيح مسلم بشرح النووي، وما بعدها قول النووي هذا وحكايته عن جماهير العلماء يقوداننا إلى خيارات؛ أولها أن نبقى على ما نحن عليه، نحرم التصوير، ونراه شديد الحرمة، وتبقى الأمة معظمها بعيدة عن القيام بهذه الفتوى والأخذ بها، ويعيش كثير من المسلمين تناقضا بين ما يدعون إليه وما يفعلونه وثانيها أن نراجع المتون، ونعيد النظر في الفتاوى، ونفتح بابا للنقد والمراجعة كبيرا وثالثها أن نسكت عن هذا التناقض، وندع الحديث فيه، فيفتي المفتي بحرمة التصوير في مكان يعج بآلات التصوير، وتزدحم فيه، ويحمل كل مسلم حضره جواله؛ كي يوثق هذا اللقاء، وينشره ما استطاع بعد ذلك في العالمين من حوله أفضل هذه الخيارات عندي، وأقلها ضررا، وأعظمها نفعا، وأجزلها عائدة علينا؛ هو الخيار الثاني الذي يفتح باب النقد والمراجعة على كل الجهود البشرية، وما بني عليها، وخرج إلينا من تحتها، وهو الدور الذي يلزمنا القيام به، والنهوض بأعبائه، وهو أيضا أصعب تلك الخيارات؛ لكنه الحل الأليق بنا وبأمتنا التي أصبح تناقضها شيئا لا يمكن ستره عن الناس ولا خفاؤه عن الشعوب هناك بون كبير بين دلالة النص الذائعة بين المسلمين في تحريم التصوير والتشديد فيه وبين واقعهم الذي يعيشونه في أرجاء المعمورة، ويسيرون عليه يذيع بين كثير من المسلمين أن التصوير حرام، ومقترفه على خطر عظيم، ويشيع بينهم التصوير وينتشر كأنه من أحل الأشياء وأكثرها إباحة، يلجأ إليه الكبير والصغير، ويوثق به المسلم، رجلا كان أم امرأة، يومياته، ويرصد به أحداث ليله ونهاره لم يغز التصوير حياتنا الاجتماعية، ولم يقف عند حدود ما يجري فيها من مناسبات وأحداث؛ بل تجاوز ذلك إلى دور العبادة وبيوت الله تعالى، ففشت الجوالات في المسجد الحرام، وانتشرت في المسجد النبوي، ورأينا جميعا المسلمين، من معتمرين ومصلين، يلتقطون الصور، ويأخذون المقاطع لهم ولمن معهم، واختفى صوت الإنكار، وغابت نبرة الواعظ، وما زالت هذه النازلة تتسع فينا دائرتها، وينداح في حياتنا مداها، وانتهينا إلى أن ننقل إلى العالم عبر وسائط النقل المباشر وغير المباشر مخالفتنا في بيوت الله لمنطوق النص في التصوير ولفتوى جماهير العلماء، فإلى متى نبقى على هذه الحالة، نقول شيئا ونسير في حياتنا على غيره؟نقلا عن الرياض جميع المقالات المنشورة
