فيأسلاكـِـهوصانعــهُالمـوتَفيبطنـِـهِحشـاوماضَحْكـــةمنثغرِسمراءَمشرقٍوكأنّهافيالحسنِحوريـّـةالسمــابأشرقَمِنْعَزْمالمجاهدِفيالوَغىفهمّتهفيعِزِّإلـــىالمجدِقــدسمــــاحامدبنعبداللهالعليالقصيـدةالمفخـّـخـّــةمِنْسِحرعَينيْهــــــاأمـَـزِّق،،ذَاتــــــِــــيفتطيــــرُأشلائـيطربــاً،،تنشــدُلحنَالذّكريــاتِتَرقـُـصُ،،فيرُبـَىالآهــــــاتْفأريجُهــا،،مِـنصَدىآهـاتيبينكلّالنّاس،،كنــــتِ،،وستبقيْننِيـليمِثْلمـاكُنــتِ،،فُراتــِيوعَبيرُروُحـيكنتِدوْمـــافَهــَـاتيأسقنيالحـــبّصِرفــاًوبِهعذّبينيواحْضِنـيلَهَفـَـاتيلـســتُأطلـــبُثغـراًثائـراً،،بالقُبــُـــــــلاتِلســـتأنشـدُوصـــلاً،،لاهبـــاً،،بالشهـــواتِلستُياعمــــــري،،كــلاّ،،لســــتُأبغيالحُرُمـاتِصحيــحٌ،،تَقْتلينــِـــــيبناعِسِالنظـــراتِكلّماشئــتِ،،ثــمّتُحيّيــنفـُــؤاديبنسيـمِالهَمَـساتُ،،بلَمسـةِالأشــواق،،ورحيــقِالضَحكــــاتِوصحيــحٌسِحرُلَحْظـكياحُبـّيأبْهـَرسحـرَالكلمـــاتِوصحيحٌحُسْـنــُــك،،الحَانــي،،يجمــعُ،،شَمْليوشــَـتاتـيوصحيحٌكـــــلّمافيـكِفوقرَسْـمالصفـاتِوصحيحٌكلّشِعريلايصفُالعُشـرَمِنـْـك،،ياحيـاتيوصحيــــحٌأنـتِمـنّيعينُذاتيبـــلْ،،سويداءُنَواتـيلكــنْ،،ليسَهذاَمَطلبـي....واحَسـَـراتي....إنماأطلبُشيئاًواحــداًواحــداًلاتّردّينيفديتُكِيافتاتيفجّريهــموامنحينــــِـيفَرْحَتــِـيبَعـْدالمماتِحامدبنعبداللهالعليقُلللكَرَىماذاجَرىودّعتجفنايُشترىبالروحيُفدىوالدُنـــَـاحتماتُباعبلامراماللمحبّةحُكمهاجارٍعلـــــىكلالورَىأتحبّنيالحسناءسرّا؟ويــــحقلبيمادرَىتلـكالجفُونتحبِّنيوالقلــبُيبكيساهـــرا؟تلكالتيكالشمسإنبرَزَتسواهالاترى؟آهٍمنحسنتألّـقَبالورودوبالزهورتعطـّراوتبسّمٍكلآلىءٍشعّتبـخـــــــدّأحمـــــــراولحُوظُهاكالسّحروالرمــشمريضافاتراوالثغرُكالحُبّأوّلنارِهِمستطيـراثائــراكُفّيفديتُكعنبكـاءِكجاءقلبيحاضرايهويإليكوينحنـيبفناءحبـّـكحائـــــرابلعـَدّهذايافتىواذْكـرشُجاعاجاســــراليثٌإنْبدَافيالحــربِجَلـْـداصابراسيمزّقالأعداءسَحْقــامُستهيـِــناقاهــــــــراأباالداهياتسرْلاعَدِمتُكَفيطريقكظافـِرابِكَلاعدِمتُكَلاســواكَسَأصطفيكَمفاخـِـراأنتَالذيبالحقّقُمـتعلىالعدوّمجاهــراوضربتَبالصمصــــامِجيشَحقدٍكافــــــرافتناثرواشذَراًوغدالواؤهُمُذليلاًخائـــراًسيسجّلالتاريخمجدَكسـوفَيُبلِغُكَالذُرىحامدبنعبداللها فيأسلاكـِـهوصانعــهُالمـوتَفيبطنـِـهِحشـاوماضَحْكـــةمنثغرِسمراءَمشرقٍوكأنّهافيالحسنِحوريـّـةالسمــابأشرقَمِنْعَزْمالمجاهدِفيالوَغىفهمّتهفيعِزِّإلـــىالمجدِقــدسمــــاحامدبنعبداللهالعليالقصيـدةالمفخـّـخـّــةمِنْسِحرعَينيْهــــــاأمـَـزِّق،،ذَاتــــــِــــيفتطيــــرُأشلائـيطربــاً،،تنشــدُلحنَالذّكريــاتِتَرقـُـصُ،،فيرُبـَىالآهــــــاتْفأريجُهــا،،مِـنصَدىآهـاتيبينكلّالنّاس،،كنــــتِ،،وستبقيْننِيـليمِثْلمـاكُنــتِ،،فُراتــِيوعَبيرُروُحـيكنتِدوْمـــافَهــَـاتيأسقنيالحـــبّصِرفــاًوبِهعذّبينيواحْضِنـيلَهَفـَـاتيلـســتُأطلـــبُثغـراًثائـراً،،بالقُبــُـــــــلاتِلســـتأنشـدُوصـــلاً،،لاهبـــاً،،بالشهـــواتِلستُياعمــــــري،،كــلاّ،،لســــتُأبغيالحُرُمـاتِصحيــحٌ،،تَقْتلينــِـــــيبناعِسِالنظـــراتِكلّماشئــتِ،،ثــمّتُحيّيــنفـُــؤاديبنسيـمِالهَمَـساتُ،،بلَمسـةِالأشــواق،،ورحيــقِالضَحكــــاتِوصحيــحٌسِحرُلَحْظـكياحُبـّيأبْهـَرسحـرَالكلمـــاتِوصحيحٌحُسْـنــُــك،،الحَانــي،،يجمــعُ،،شَمْليوشــَـتاتـيوصحيحٌكـــــلّمافيـكِفوقرَسْـمالصفـاتِوصحيحٌكلّشِعريلايصفُالعُشـرَمِنـْـك،،ياحيـاتيوصحيــــحٌأنـتِمـنّيعينُذاتيبـــلْ،،سويداءُنَواتـيلكــنْ،،ليسَهذاَمَطلبـي....واحَسـَـراتي....إنماأطلبُشيئاًواحــداًواحــداًلاتّردّينيفديتُكِيافتاتيفجّريهــموامنحينــــِـيفَرْحَتــِـيبَعـْدالمماتِحامدبنعبداللهالعليقُلللكَرَىماذاجَرىودّعتجفنايُشترىبالروحيُفدىوالدُنـــَـاحتماتُباعبلامراماللمحبّةحُكمهاجارٍعلـــــىكلالورَىأتحبّنيالحسناءسرّا؟ويــــحقلبيمادرَىتلـكالجفُونتحبِّنيوالقلــبُيبكيساهـــرا؟تلكالتيكالشمسإنبرَزَتسواهالاترى؟آهٍمنحسنتألّـقَبالورودوبالزهورتعطـّراوتبسّمٍكلآلىءٍشعّتبـخـــــــدّأحمـــــــراولحُوظُهاكالسّحروالرمــشمريضافاتراوالثغرُكالحُبّأوّلنارِهِمستطيـراثائــراكُفّيفديتُكعنبكـاءِكجاءقلبيحاضرايهويإليكوينحنـيبفناءحبـّـكحائـــــرابلعـَدّهذايافتىواذْكـرشُجاعاجاســــراليثٌإنْبدَافيالحــربِجَلـْـداصابراسيمزّقالأعداءسَحْقــامُستهيـِــناقاهــــــــراأباالداهياتسرْلاعَدِمتُكَفيطريقكظافـِرابِكَلاعدِمتُكَلاســواكَسَأصطفيكَمفاخـِـراأنتَالذيبالحقّقُمـتعلىالعدوّمجاهــراوضربتَبالصمصــــامِجيشَحقدٍكافــــــرافتناثرواشذَراًوغدالواؤهُمُذليلاًخائـــراًسيسجّلالتاريخمجدَكسـوفَيُبلِغُكَالذُرىحامدبنعبداللها
