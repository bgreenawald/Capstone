الأختنورةأحتـــرمُوجْهـــةنظركِالخاصّةبسبــــبإغلاقِالموضوعْ،غيـــْرأنيلاأتحمّلردودغيـــْري،كماأنّكِلاتتحمّــــليمسؤوليةالغبـــَـاءوالسخــــافَةوالعقْليــــّةالخرْفانيّــــةلمرتاديشبكةالطحــــالب،أوأرضِالمـــوزْ،كمـــاأنهليْسمنمسؤوليــــتيأنأتدخّلفيسيـــاسةالإدارةماإذاكانهنــــاكشخصمايمــــونعليهاأملاوبالمناسبَــــةفهوليسابنعمّـــي،ولوقـــارنّاسياســــةالمنتَدياتلقابلتنــــاسياسةالحسبــــةبنفسالأســــلوبلمنتقوللهمالإدارةدومـــاًتِكـــْرَمعيونـــــكوتسمحلهمبالسبوالشتــــم،بلوتثبّــــتمواضيعهــــمأيضاً،تفسيـــرِكلسببطرحكِللموضــــوعِمنطقيجدّاً،والماكيــــنَةالعربيّةأكيـــدماتقصّر،وستُقـــابِلكلمنيطعَـــنفيطالبـــانالعرببِجُمـــلةالترحيبالشهيــــرةيااللـّهنتحيّيــــهم،أسفرتوأنــــورتسيّمــــالوكانتطالبــــانالعربخليجيّـــة،لكنّــــيأظنأنتجربةطالبانالعــــربوالتيفشلتفيأنتُحقّــــقأدنىأهدافها،وهوالوصــــوللقلبالمسلمالعربــــيوإثارةقضايـــاهوالدفاععنه،وإذبهـــــاتتوقععلىنفسهــــاوتنطلِـــقلِتُهاجِــــممنلأجلهِمخرجت،فخســــرتتأييداشعبيــــاًثمّةمئاتالمؤشـــراتكانتتدلعلىمدىتغلغلــــهفيأوساطِالناسالعـــوَام،وأحسَـــبُأنهلوتمّاستثمـــارَهبشكلناجِحلرأينــــا،علىالأقـــل،وجهجزيرةالعربقدتغيّــــرحتماً،لكنّالفشــــللاحقهامنمرحلةٍإلىمرحــــلة،فخسرتآخرأوراقــــهاوهيالتأييدالشعبــــي،ثممالبِثتأنمــــاتت،والتبــــريرجاهزطبعاًَبأنهممثلَأصحــــابُالأخدود،أمّـــاسببتهكّميعلىأكثريّةالعرب،فيكْفـــيأنأحيلكِإلىأسواقمارينــــامولفيالسّـــالميّة،أوتمشيأمتارقليـــلةبجانبالماريناإلىمطعمجـــراندكافيه،أوتدخلـــيحديقةالشعبأوتروحيـــنمنتزهفيلجةالسيــــاحي،أوتتمشيــــنفيشـــارعالخليجوتميلينعلىمطعمفـــرايديز؛وفيهـــذهالأماكِنوغيرهايُمكِنـــكِأنتُشاهديعيّنةمنالعرب،وكثيـــرمنهمعلىذلك،الذينتأمليـــنفيأنيُشكّـــلواطالبانالعــــرَب؛أظنّوقتهــــاستتأكّدينأننجومالظهـــروليستنجومفندقالســــالميّةبالاسهيأقــــربليمنأنأحـــلمبخروجطالبـــانالعربوالأمّــــةلاتُعدمالخيـــْربالطّبـــع،وليْـــسَمنشيءٍأفرحلقـــلبيمنمُشاهدةطالبانالعــــرب،فلستُمثبّطاًمخذّلاًداعيــــاًلليأس،ولكنـــيداعٍإلىالتعلّممنالأخطــــاء،ومنلايخطألاينجحْ،واستثمــــارِهابشكلأفضلبحيثتؤتــــيالتجاربالجهاديةثمارهــــا،وتؤتــــيأكلها،سائلــــينالمولىع
