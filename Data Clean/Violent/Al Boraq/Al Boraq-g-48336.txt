اقتباسكاتبهذهالمشاركةعامرالجراحواللهانيتعجبتمنانبعضالاعضاءفيحنينللاسفانساقواخلفهذهالدعايةحسبنااللهونعمالوكيلكلامكرائعوتكلمتبنصفمااردتاناتكلمانابه،لكنمنتظرلحظةالانفجاروهيقريبةجداعلىمااظن،لانيباختصارطاكصورلكثيرمنالاعضاءفيقلبيوبعضهمعملتلهمتماثيل،لكننيلااريداناشقماباتيسمىبانصارالجهاد.اللهالمستعانأخــيعامر،لــنتختلفالعقليــّاتفيحُنيــْنعنغيــرِها،فـالخرفــانخرفانأينــماحلّوا،ولودخلتَالساحاتالعربيــّةلوجدتهمهناكأيضاً،فليستسُمــعَةالمنتدىبمانِعــةٍمنتواجدهم،وإنكنتأرىأنالعقــَلاءفيحُنينأكثــر،أذكــرأنهذهالحــالةمّرتمنقبلعلىمنتدياتالخرفــانتلكحينماظهــرترسالةراكانبنوليامزالجــنديالسّريالمفترضللقاعدةفيأمريــكا،والتيتلــتخطاباًللشيخأســامةيوضّحفيهخلفيــاتالحربالقائــمةحالياً،فجرىهؤلاءكعادتهمخلــفتطبيلهذاالـراكان،وماذاكانتالنتيجة؟عـــادواليناموانومأهلالكهــف،السببالرئيســيباعتقاديهوعــُقدةالنقصالمصاببهاكلالخــرفان،النقصالناشــئعنعدكالتحرّكعلىأرضالواقعبشــيء،والكبــتالذييُلاقونهمنحكومــاتدولهم،فيكــونالإنترنتوالأفلامالقرعةهذهمُتنفّــسهمالوحيد،فالـــكليُريدمنالقاعدةأنتتدخلفيبلــده،حتىأعجبنيتعليقأحدهممــرّةعلىأحدالذيندعــوْالدخولالقــاعدةبلدهمارأيكَأنتتدخلالقــاعدةلحلّمشاكــلكَمعحماتكبسببآخــرطبخَــةلمتتُقِنــهازوجتــك؟ســأكونأسعدالناسلوضُرِبتأمريــكامرّةأخرى،أوالكيانالعــبري،وليسمعنــىأننُهاجِمهؤلاءالخرفــانأننشطبالإنصــافمنقاموسنا،بلالإنصافمطلوب،فيالرضــاوالسخــط،ولكنأنتتمالأمــوربهذاالشكلالبهلوانيالتهريجــيفهذامايحتــاجلوقفةقويــّة،أبشــرك،فقدصاروايحــللونالشيــفرةالكامنةفيمعــرّفالشخصالذيطرَحالموضــوع،واستنتجأحــدالبُلــهَاءأنالضربةالقادمةستكــونبـالطاعــون،يُذكّــرنيذلكبالشــيفرةالسريّةالتياكتشفهانصيــرالحق،أحدخرفانأرضالمــوز،فيإحدىخطابــاتالشيخأسامة،
