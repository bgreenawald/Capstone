ورقات في تحذير المقاومة من الارتداد كالصحوات ومن الناس من يعبد الله على حرف فإن أصابه خير اطمأن به وإن أصابته فتنة انقلب على وجهه خسر الدنيا والآخرة ذلك هو الخسران المبين الحج بسم الله الرحمن الرحيم اعلم رحمك الله تعالى أن من سنة الله في دعاة التوحيد، معاداة الناس لهم، وتكالب أهل الكفر والنفاق عليهم، بالاستهزاء والافتراء، وبالحرب والاعتداء. وليس هذا خاصا بالأنبياء وحدهم، فكما قيل عن محمد صلى الله عليه وسلم هذا ساحر كذاب ، وأخرج من دياره، وتداعى عليه وعلى أصحابه كفار زمانه كتابيهم ووثنيهم، فإن من ورث عنه دعوته، قوبل بمثله بالتكذيب والاستهزاء والافتراء، وحاربه أهل الشرك والنفاق، فما زال أهل الجاهلية يتوارثون عن بعضهم العداء لأهل التوحيد، وما زال دعاة التوحيد وأهل الجاهلية فتنة لبعض وابتلاء، كما قال تعالى وجعلنا بعضكم لبعض فتنة أتصبرون وكان ربك بصيرا . واعلم أنك إن تمسكت بدينك وأعظمه أصله وهو التوحيد ، وعضضت بنواجذك على سنة نبيك صلى الله عليه وسلم وأعظمها سنته في معاداته للكافرين ، فإن الله مبتليك ومعرضك للفتنة؛ ليراك وهو أعلم بك هل أنت من الصادقين في إيمانهم، أم من أصحاب الدعاوى الكاذبة التي يظهر زيفها عند أول امتحان، قال تعالى أحسب الناس أن يتركوا أن يقولوا آمنا وهم لا يفتنون ، ولقد فتنا الذين من قبلهم فليعلمن الله الذين صدقوا وليعلمن الكاذبين ، وقال تعالى أم حسبتم أن تدخلوا الجنة ولما يأتكم مثل الذين خلوا من قبلكم مستهم البأساء والضراء وزلزلوا حتى يقول الرسول والذين آمنوا معه متى نصر الله ألا إن نصر الله قريب ، وقال تعالى وليمحص الله الذين آمنوا ويمحق الكافرين ، فأما أهل الصدق والإيمان، فيثبتون ويصبرون ويجاهدون أهل الكفر، وأما أهل النفاق ومن لم يتمكن الإيمان من قلوبهم، فيفتنون وينقلبون على أعقابهم، ويسارعون إلى الكفار ليتخذوهم أولياء من دون المؤمنين، كما في قوله تعالى أم حسبتم أن تتركوا ولما يعلم الله الذين جاهدوا منكم ولم يتخذوا من دون الله ولا رسوله ولا المؤمنين وليجة والله خبير بما تعملون . قال ابن جرير الطبري عند قوله وليجة وإنما عنى بها في هذا الموضع البطانة من المشركين. نهى الله المؤمنين أن يتخذوا من عدوهم من المشركين أولياء، يفشون إليهم أسرارهم والله خبير بما تعملون ، يقول والله ذو خبرة بما تعملون، من اتخاذكم من دون الله ودون رسوله والمؤمنين به أولياء وبطانة، بعد ما قد نهاكم عنه، لا يخفى ذلك عليه، ولا غيره من أعمالكم، والله مجازيكم على ذلك، إن خيرا فخيرا، وإن شرا فشرا. وبنحو الذي قلت في معنى الوليجة ، قال أهل التأويل . فإياك إياك أن تكون ممن تزيغ الفتن قلوبهم، فتفعل فعل أهل النفاق، الذين تظهرهم الفتن، وتكشفهم المواقف، ويتساقطون عند الابتلاءات. فإن الناس في الفتنة كما علمت ثابت على الحق متجمل بالصبر، أو هالك مسارع في أهل الكفر، معتذر لنفسه بأقبح العذر. قال الشيخ حمد بن عتيق رحمه الله قلت وما ذكره من الافتتان قد رأينا ما هو نظيره، أو أعظم منه في هذه الأزمان، وكذلك انقسم الناس أقساما أحدها ناصر لدين الإسلام، وساع في ذلك بكل جهده ، وهم القليلون عددا، الأعظمون عند الله أجرا. القسم الثاني خاذل لأهل الإسلام تارك لمعونتهم. القسم الثالث خارج عن شريعة الإسلام بمظاهرة حزب المشركين ومناصحتهم. وقد روى الطبراني عن ابن عباس عن النبي صلى الله عليه وسلم قال من أعان صاحب باطل ليدحض بباطله حقا فقد برئت منه ذمة الله وذمة نبيه . واعلم جنبك الله سبيل من أهواه هواه أن من حالف الكافرين، وناصرهم على المسلمين؛ فقد ارتد عن الدين، وصار كافرا بالله العظيم. يستوي في ذلك من ناصرهم بقتال المسلمين معهم، أو قتال المسلمين بأمرهم وبالتنسيق معهم، أو بالدلالة على عورات المسلمين، وأماكن تواجد المجاهدين، وبيوتهم، ومستودعات أسلحتهم، أو بكلمة تضر بالمجاهدين قالها نصرة للصليبيين، أو طاعة لأمرهم. ويستوي في ذلك من ناصرهم على المسلمين طامعا في الدنيا من مال ومنصب وحماية، أو خائفا من بطش الكافرين متقيا شرهم، ولو كان مبغضا لهم، معتقدا الإسلام، كل هؤلاء يكفرون بمناصرة الكفار على أهل الإسلام. وهذا من أوضح الأحكام في كتاب الله وسنة رسوله صلى الله عليه وسلم، ومما أجمع عليه أئمة الدين، وتوارد على تقريره علماء الشريعة. وقبل ذكر دلائل هذا، أقوم بتقرير قاعدتين فيهما رد عام على من أحدث شرط الاعتقاد لتكفير من ناصر الكفار وظاهرهم على المسلمين، لإبطال شرطهم هذا. القاعدة الأولى أن شروط الأحكام وضعية من الشارع، والأدلة على كفر من ظاهر الكفار مطلقة لم تقيد بالاعتقاد، فاشتراط الاعتقاد استدراك على الشارع. والقاعدة الثانية أن الاعتقاد المذكور على تنوع صوره هو كفر مستقل، من وجد منه كفر، ولو لم يظاهر على المسلمين، فلا يكون للمظاهرة تأثير في الكفر على التحقيق ، فيكون الشارع قد علق الأحكام بأوصاف لا تأثير لها ، وهذا عبث ينزه عنه الشارع. وبيان ذلك من وجهين أحدهما أن الشرط عند الأصوليين هو ما لا يوجد المشروط مع عدمه، ولا يلزم أن يوجد عند وجوده ، وشرط الكفر عند هؤلاء هو الاعتقاد، والمشروط هو الكفر، ويلزم من وجود الاعتقاد وجود الكفر، فلا يكون شرطا، بل هو سبب للكفر مستقل بنفسه. والوجه الثاني هو أن الحكم الكفر في الآيات معلق بوصف هو الموالاة ، قال القرافي وترتيب الحكم على الوصف يدل على علية ذلك الوصف لذلك الحكم وعدم علية غيره ، فدل أن علة الكفر هي الموالاة والمناصرة نفسها لا غيرها، وتقدم بيان بطلان جعل الاعتقاد شرطا، فالاعتقاد و المناصرة كلاهما سبب مستقل بنفسه للكفر، يكفر من وجد منه ولو واحد منهما. وإليك شيء من الدلائل على ما تقدم تقريره ، مرتبة بذكر الدليل، ثم ذكر مسائل عليه، في الدلالة على كفر من ناصر الكفار على المسلمين، وفي الدلالات الخاصة على بطلان شرط الاعتقاد لتكفيره، وغير ذلك مما يستفاد من الأدلة. الدليل الأول قوله تعالى يا أيها الذين آمنوا لا تتخذوا اليهود والنصارى أولياء بعضهم أولياء بعض ومن يتولهم منكم فإنه منهم إن الله لا يهدي القوم الظالمين ، فترى الذين في قلوبهم مرض يسارعون فيهم يقولون نخشى أن تصيبنا دائرة ... الآيات . وفي الآيتين مسائل المسألة الأولى في الآيتين دلالة على كفر من ناصر الكفار على المسلمين من وجوه أحدها أن قوله بعضهم أولياء بعض فيه أن من والى بعض الكفار فهو من بعضهم، أي كافر مثلهم. والثاني قوله فإنه منهم ، أي من جملتهم، فهو كافر. والثالث قوله القوم الظالمين أي الكافرين؛ فالظلم إذا أطلق معرفا فالأصل فيه هو الكفر الأكبر ، لقوله تعالى والكافرون هم الظالمون ، وعن ابن مسعود رضي الله عنه قال لما نزلت الذين آمنوا ولم يلبسوا إيمانهم بظلم شق ذلك على أصحاب رسول الله صلى الله عليه وسلم ، وقالوا أينا لا يظلم نفسه؟ فقال رسول الله صلى الله عليه وسلم ليس هو كما تظنون، إنما هو كما قال لقمان لابنه يا بني لا تشرك بالله إن الشرك لظلم عظيم . والرابع قوله في قلوبهم مرض وهو النفاق، فكفرهم بذلك . المسألة الثانية فيها بطلان شرط الاعتقاد لتكفير من ناصر الكفار، من وجوه أحدها قوله ومن يتولهم علق الحكم بالكفر على الولاية والنصرة؛ فدل أنها هي سبب الكفر . الثاني قوله فترى . والثالث قوله يسارعون فيهما أن الكفر الذي وقعوا فيه عمل ظاهر، وليس اعتقادا باطنا . الرابع قوله نخشى اعتذروا بالخوف ولم يكذبهم الله تعالى. وكفروا مع أن مناصرتهم لم تكن من اعتقاد بل من الخوف . الخامس قوله أن تصيبنا دائرة فيه أنهم وقعوا فيما كفروا به ليحموا أنفسهم من الكفار، وحماية النفس تكون بالأعمال الظاهرة لا بالاعتقاد الباطن، فدل أن ما كفروا به هو أمر ظاهر وهو المناصرة. المسألة الثالثة فيها أن من وقع في موالاة الكفار ومناصرتهم على المسلمين خوفا من تسلطهم عليه وبطشهم به فإنه يكفر ولا يعذر بخوفه، فيكفر من ناصرهم طمعا في المال والمنصب من باب أولى بمفهوم الموافقة للآية. الدليل الثاني قوله تعالى ترى كثيرا منهم يتولون الذين كفروا لبئس ما قدمت لهم أنفسهم أن سخط الله عليهم وفي العذاب هم خالدون ، ولو كانوا يؤمنون بالله والنبي وما أنزل إليه ما اتخذوهم أولياء ولكن كثيرا منهم فاسقون . وفي الآيتين مسألتان المسألة الأولى في الآيتين دلالة على كفر من ناصر الكفار على المسلمين من وجهين، أحدهما قوله خالدون والخلود في النار للكفار. الثاني قوله ولو كانوا يؤمنون بالله والنبي وما أنزل إليه ما اتخذوهم أولياء فيه أن الإيمان بالله وموالاة الكفار ضدان لا يجتمعان. المسألة الثانية فيها بطلان شرط الاعتقاد لتكفير من ناصر الكفار، فقوله ترى فيه أن الكفر الذي وقعوا فيه عمل ظاهر، وليس اعتقادا باطنا. الدليل الثالث قوله تعالى لا يتخذ المؤمنون الكافرين أولياء من دون المؤمنين ومن يفعل ذلك فليس من الله في شيء إلا أن تتقوا منهم تقاة ويحذركم الله نفسه وإلى الله المصير . وفي الآية مسائل المسألة الأولى في الآية دلالة على كفر من ناصر الكفار على المسلمين، فقوله شيء في قوله فليس من الله في شيء نكرة في سياق النفي، فدل على العموم، وفيه براءة الله منه مطلقا، وذلك هو الكفر. المسألة الثانية فيها بطلان شرط الاعتقاد لتكفير من ناصر الكفار، فقوله إلا أن تتقوا منهم تقاة فيه رخصة في الوقوع في الكفر للتقية، و الاعتقاد شيء باطن لا تقية فيه، فدل أن الكفر المرخص فيه عند التقية هو أمر ظاهر. المسألة الثالثة التقية المرخص بها ليست على الإطلاق، وإنما هي في حال الإكراه ، بين ذلك قوله يقولون نخشى أن تصيبنا دائرة وتقدم الكلام على ذلك. الدليل الرابع قوله بشر المنافقين بأن لهم عذابا أليما ، الذين يتخذون الكافرين أولياء من دون المؤمنين أيبتغون عندهم العزة فإن العزة لله جميعا . وفي الآيتين مسائل المسألة الأولى في الآيتين دلالة على كفر من ناصر الكفار على المسلمين، لقوله المنافقين فقد عقب الحكم بالنفاق بالموالاة، وهذا من أساليب التعليل، فدل على أن الموالاة سبب النفاق والكفر. المسألة الثانية فيها بطلان شرط الاعتقاد لتكفير من ناصر الكفار، لقوله أيبتغون عندهم العزة فهم كفروا لابتغاء العزة ، والعزة تطلب بالعمل الظاهر لا بالاعتقاد الباطن، فدل أنهم كفروا بعمل ظاهر، هو المناصرة. المسألة الثالثة دل المنطوق في الآية على أن ابتغاء العزة ليس عذرا في موالاة الكفار، فمن ناصرهم على المسلمين لتحصيل المطامع فإنه يكفر بذلك. وتقدم الاستدلال عليه بمفهوم الموافقة لآية سابقة. المسألة الرابعة فيها تبشير للمؤمنين، وتحذير لضعاف النفوس، بأن العزة لله جميعا، فلا يضع العاقل دينه ابتغاء العزة، فيخسر دينه ودنياه، ويرث الذل أبدا. الدليل الخامس قوله إنما ينهاكم الله عن الذين قاتلوكم في الدين وأخرجوكم من دياركم وظاهروا على إخراجكم أن تولوهم ومن يتولهم فأولئك هم الظالمون . دلت الآية على كفر من ناصر الكفار على المسلمين، لقوله ومن يتولهم فأولئك هم الظالمون وإذا أطلق الظلم يراد به الكفر الأكبر ، وتقدم الكلام في ذلك. الدليل السادس قوله ثم أنتم هؤلاء تقتلون أنفسكم وتخرجون فريقا منكم من ديارهم تظاهرون عليهم بالإثم والعدوان وإن يأتوكم أسارى تفادوهم وهو محرم عليكم إخراجهم أفتؤمنون ببعض الكتاب وتكفرون ببعض فما جزاء من يفعل ذلك منكم إلا خزي في الحياة الدنيا ويوم القيامة يردون إلى أشد العذاب وما الله بغافل عما تعملون ، أولئك الذين اشتروا الحياة الدنيا بالآخرة فلا يخفف عنهم العذاب ولا هم ينصرون . وفي الآيتين مسائل المسألة الأولى في بيان معنى الآيات بإيجاز، روى الطبري عن ابن عباس، قال ثم أنتم هؤلاء تقتلون أنفسكم وتخرجون فريقا منكم من ديارهم تظاهرون عليهم بالإثم والعدوان إلى أهل الشرك، حتى تسفكوا دماءهم معهم، وتخرجوهم من ديارهم معهم. قال أنبهم الله على ذلك من فعلهم، وقد حرم عليهم في التوراة سفك دمائهم، وافترض عليهم فيها فداء أسراهم . المسألة الثانية فيها دلالة على كفر من ناصر الكفار على المسلمين، وذلك من وجوه أحدها قوله وتكفرون ببعض فجعل مناصرة الكفار على المسلمين كفرا ببعض الكتاب، وهو كفر به. الثاني قوله ويوم القيامة يردون إلى أشد العذاب دخلت أل على العذاب، فالمراد جنس العذاب، وأشده يكون للكفار. الثالث قوله اشتروا الحياة الدنيا بالآخرة فيه خسرانهم الآخرة، فلا حظ لهم فيها. الرابع قوله فلا يخفف عنهم العذاب فيه دوام العذاب وخلودهم فيه. الخامس قوله ولا هم ينصرون فيه دوام العذاب عليهم، وعدم انقطاعه بقوة أحد أو شفاعته. المسألة الثالثة فيها بطلان شرط الاعتقاد لتكفير من ناصر الكفار، لقوله وتخرجون فريقا منكم من ديارهم تظاهرون عليهم فعلق الحكم عليهم بالكفر على الإخراج والمظاهرة، وفي قوله فما جزاء من يفعل ذلك علق الجزاء بالكفر على الفعل. المسألة الرابعة قوله أولئك الذين اشتروا الحياة الدنيا بالآخرة فيه أن من وقع في الكفر من أجل الدنيا فليس بمعذور، بل يكفر ويخسر الآخرة، وتقدم مثل ذلك. المسألة الخامسة في قوله أفتؤمنون ببعض الكتاب وتكفرون ببعض تحذير لمن يخلط عملا صالحا وآخر سيئا، فيقاتل الكفار في مواضع وأحوال، ويناصرهم على المسلمين في مواضع وأحوال أخرى، وأن هذا ليس بسالم من الكفر. الدليل السابع قوله تعالى وقد نزل عليكم في الكتاب أن إذا سمعتم آيات الله يكفر بها ويستهزأ بها فلا تقعدوا معهم حتى يخوضوا في حديث غيره إنكم إذا مثلهم إن الله جامع المنافقين والكافرين في جهنم جميعا . وفي الآية مسألتان المسألة الأولى قوله إنكم إذا مثلهم فيه كفر القاعد المستمع للمستهزئين بآيات الله بغير إنكار، وكونه بمنزلة الفاعل، وإذا كان القعود وعدم الإنكار كفرا ولو لم يشاركهم الاستهزاء، فنصرة الكفار على المسلمين وإعانتهم على ما فيه ظهور للكفر هو كفر من باب أولى مفهوم الموافقة . المسألة الثانية فيها تحذير لمن يتهاونون في تمام البراءة ممن يناصرون الكفار، ويعتذرون لهم بالأعذار، فترك إنكار المنكر من أبواب الوقوع فيه. وبعد، فلا يغرنك مشايخ شريعة الشيطان، ورثة إمامهم ومجدد ملتهم بلعم بن باعوراء ، واحذر سلوك طريقهم، فهو الردى في الدنيا والردة عن الدين.
