ينام أخي على زنــــدي ... أظــــــلله بأهدابـــــــي وفي قلبي فرشــــت لـه ... فهل يدري أخي مابي؟ بحثت بوجهــه الدامــــي ... لأزرع قبـــــــــلـــة فيــه فحيّــــرني وآلمنـــــــي ... بتمثيـــــل وتشويــــــــهِ وفارقنــــي بلا دمـــــع ... وأبلغ منــــه ذا الصمـــتُ وأكظم غضبتي حينــــا ... لكل تفجـــــــــرٍ وقـــــتُ أكاد أصيح من حَزَنـــي ... بأن فؤادي احترقــــــــا سأصبر إن في صــدري... وان غالبتُه الرهقـــــــــــا أتوكَ أخي بما ملكَــــتْ ... حضـــــارتهم من القهــــرِ وحين سقطتَ لم يجـــدوا ... رصاص الغدر في الظهــرِ فأنت الصــامد البطـــــلُ ... وأنت بدربنـــــا مثـــــــلُ ثبـت لهم ولم تهـــــــــزم ... وحاشا يهــزم الجبـــــــلُ وحيــــدا قاتـــل الاعــدا ... و ردّ جمــــــــــوعهم ردّا ثلاث قذائـــــفَ انفجــرت ... و ما فتــت له عضــــــدا ومن أوكارهم خرجـــــت ... رصاصــات بها الوهــــنُ و لكن كـــــل مخـلــــوق ... بيـــوم المـوت مرتـــــهنُ فلم يسمـــع له صــــــوت... يقـــول بأننـــي مـــــــتُّ ولكن صـــــاح في فـرحٍ ... بإحداهـــــن قــد فــزتُ فواعجبـــا لمن يقضــــي ... يظــل سلاحــــــه معـــه و يــأبى أن يفـــــــارقه ... و يخشى أن يـــودعـــه فويــــــــل للعلوج الحمر ... من غضبــي ومن ثــاري سـأحصـدهم بمنجلـــــهم ... و أقذفـــهم إلى النـــار و إســــألهم إذا ضجّـــت ... رعـود الحق من أقـــوى و مَن طلقــــــاته ارتجفت ... و مَن طلاقـــــاته نشوى فيــــا عبّـــــاد أكــتوبر ... لقانـــا في ثـرى لوجــرْ صهيل الخيل في صدري ... وفي قلبي الردى زمجرْ و بــــاطلكم ســــأصرعه ... و لـو وحدي سأدفعـــه و لا ينجيـــــــــكمُ منّــي ... حديــــد الأرض أجمعـــه سأحمـــل بعدك الرشاش ... زيّـــــنه الـدم القـــــاني و أعبـــــر كــل ميـــدان ... به تكبيــــــر إخوانــــي و حين أجيء للأقصـــى ... كما يوماً أخي أوصــى سأدفع عندها الرشــاش ... يجعله الأميــر عصـــــــا يظـــل عليه متكـــــــــئاً... ليعلـــن من ربـى القـدسِ رجعنــا اليوم للأقصـــى ... فهبـّـــــوا نحو أنــدلــسِ اللهم اعد لنا مجدا اضعناه
