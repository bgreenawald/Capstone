(( الذكرى تنفع المؤمنين )) إنَّ الـحياة مـسـلسـلٌ وتـزولُ = في وَمْـضَـةٍ فستــــــــــارُها مسدولُ كـم مـيـتٍ شَـيَّـعْـتــَهُ وحَمـَلـتـَه ُ = لا بُـدَّ نَـعْـشُـكَ فـي غدٍ مَحمـــــولُ لــو كنتَ تعلمُ أنّ مــوتك قــد دنــا = مـاذا بـربك فـاعلٌ وَتقــــــــولُ؟ لا بُدّ تَضْرَعُ مسرعاً متـوسلا ً = يــــارب تُبْتُ فهل اليك سبيــــــــلُ؟ ألانَ تُبتَ وقبل ذلك غـــــافلٌ = عجباً لِمَ التسويـــفُ والتأجيـــــــــلُ؟ كم أنت تـعصي ما سَمِعْتَ لناصح ٍ = مـاذا تـظـن أفِـعْـلُـكُـمْ مـقـبـولُ؟ ذو الدين يُصْغي للنصيحة عاملا ً = أمّـا الـشـقـي فـلـلـعـنـاد يـمـيـلُ كم من دروس في الحياة مضيئة = لـكـنّ عـشـاق الـدروس قـلـيـلُ كم شاعرا فـَهِمَ الحياة بشعره ِ؟ = فالـشـعـر فـي بحر النساء يجولُ كـم حاكـمـا فهم الـحياة بحكمهِ؟ = فـالـحـكـم فـي هـذا الزمان عليلُ انـظـر نـساء أو رجـالا هَـمُّـهُمْ = مُـتـَـع الـحـيـاة فَـفَـهْـمُهـُمْ مشلولُ إنَّ الـحـيـاة تـعــــاون ومـحبة ٌ = وتـعـــــامـل ٌ وتـسـامـحٌ وأصــولُ وتنافسٌ في الخير دون تحاسدٍ= إنَّ الـحـسـود بــِغِـلـِّهِ مــقـتـــــــولُ والـحـكـم بـالـقـرآن فيها زيـنـة ٌ = أحـكـامُـهُ الـتـحريـمُ والـتـحـلـيلُ والعيش في هذي الحياة بعزَّة ٍ= فـالـذل فـيـهـا مـؤلـم وثـقـيــــــــلُ من عاش في الدنيا جميلا فِعْلـُهُ = فـَمَـقـامُـهُ يـوم الـحـسـاب جـميلُ الـحـكـم بـالإسـلام ها هو مقبل ٌ = الـنـصـر فـيـه وَبــِرُّهُ مـشـمـولُ فـيه اقتصادٌ لا يـُضاهى خيرُهُ = مـالُ الـحلال مـبـارك وجــلـيــــــلُ أين الذي فـَهـِمَ الحياة َأصولها = هــذا هـو الـمـطـلـوب والـمـأمولُ قد عاش فيها بالعبادة مخلصا ً= يـرجـو الإلـه وسـيـفُـهُ مـسـلـــولُ لا يـنـحـنـي إلا يـكـون مُصليا ً = لـولا الـصـلاة فـليـس مـنه نـُزولُ الـخـير والإصلاح هذا زرعُهُ = ورضـا الإلـه وحـبـه الـمـحصـــولُ طيــبُ الحيـــاة لمؤمن وكرامة ٌ = وَعَـدَ الإلــهُ ووعْــدُهُ مـفـعـــــولُ إنِّـي أذَكـّــرُ مـن يـريــــد تذكرا ً= إن الــحـــيـاة تـَـزَوُّدٌ ورحـيـــــــلُ هل كنتَ ترضى أنْ تكون مسافراً = لا مالَ تملكُ والطريقُ طويـــلُ؟ يا طـائـعـاً لـلـنـفـس ِمُـتَّبِـعَ الهوى = تنسى الـقـيامة بالدُّنى مشغولُ؟ هـل هـذه الـدنـيـا تـدوم لـواحـدٍ = فاغْـنَـمْ حـيـاتــك فـالـحياة أفــــولُ فـغـدا تـلاقـي الله دون تـَـزَوُّدٍ = تـبـكـي هـنـاك ولا يــُفــيـدُ عـويـلُ إنّـي أذكِّـرُ أمَّـتـي فـي ديـنها = الــديــن نــور والــظــلام بـديــــــلُ ان كان لا شمسٌ ولا قمرٌ يُرى = كيف الحياة ولا يُـرى قـنديــــــلُ؟ حرب ُالصليب تـَجَـدَّدَتْ أحداثها = قــُمْ لـلـشـهـادةِ إنـهـا إكـلـيــــــلُ الـدين أغْـلـى بل أساس ُوجودنا = إن غـاب عـنـا فـالـجـمـيع ذليـلُ والقدس نذكرها بكل مرارة ٍ= مـسـرى الـحـبيبِ دلـيـــلـُهُ جبريــــلُ خير البلاد وشر محتل بها = هـو قـــاتـل هـو ظــــــالـم ودخـيـــــلُ وتـقـول لـلمليار هل من منجدٍ = أولـيـس فـيــــكـم قـائـد مـسـؤولُ؟ أنـتـم كـثـيــر لـيس فيكم نـخـوة ٌ = هــُمْ قـِـلـة ٌ مـنـبـوذة ٌوفـلــــولُ من كان ذا سَقَمَ ٍأيـــقعدُ يــائسا ً؟ = الـيـأس داء مُـحـبـط وَوَبـيــــلُ هـذا هـو الـقـرآن فـيه علاجكم = مـهـمـا سـألـْـتُـمْ سُؤْلـُكُمْ محلــولُ يـا قدس صبراً إنَّ جند َمحمدٍ = آتـــون لـكـنَّ الـجـمـيـــعَ عَـجـــول ُ لن يخلف الله العظيم ُ قدومَهُمْ = الـوعـد مـنـه كـذلـك الـتـسـهـيـــلُ فـَـلـْتُفْـتـَحْ الأبـواب ُعـنـد دخولهمْ = وَلـْـيُـسْـمَـعْ الـتـكـبيرُ والـتهليلُ
