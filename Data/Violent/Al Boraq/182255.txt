بسم الله الرحمن الرحيم السلام عليكم ورحمة الله وبركاته ( الله سيّدَها) مـاذا أقـول لعصبـة الفـجّـارمن دنّسوا عرض النبي المختـارمن كفّروا صحب الرسول وحرّفواالقرآن واجتـرأوا بغيـر ذِمـاربئساً لهـم أو يقذفـون بفريـةٍزوج الرسول وإبنـة الأخيـار ؟بئساً لهـم أيجاهـرون بإفكهـموالله سيَّدهـا علـى الأطهـار ؟تبـاً لأحفـاد المجـوس فإنّهـممردوا على الكذب البواح العـارلم يصدقوا يوماً وكـلّ حديثهـممحضِ إفتراءاتٍ ورهـجِ غبـارجعلوا أدلّتَهـم علـى أهوائهـمفتباينـوا بتعـاقـب الأطــواروبنوا عقيدتَهم علـى أحقادهـمفهوت عقيدتُهـم بجـرفٍ هـارما يدّعون مـن الأدلـة مخجـلٌكروايـة لحمـار إبـن حـمـارزعمـوا محبتهـم لآل محـمـدٍكي يخدعوا الضعفاء في الأمصاروالله ما سلم الرسول مـن الأذىوتقوّلوا أن بعضه فـي النـار !والله مـا أعطـوا عليـاً قـدرهبل أسرفوا في وصف ذي الفقّـارعـذراً أميـر المؤمنيـن فإنّهـمأهـل الضـلال ومنبـع الفجّـاروالله لن يرضى الحسين بفعلهـمإذ أشركـوا بالواحـد القـهّـارجعلوا علياً و الحسين و غيرهمنـداً لـرب العالميـن البـاريتباً لهم أو يفرحـون بشركهـمأو لا يخافوا غضبـة الجبّـار ؟أو لا يخافـوا أن ينـزّل أمـرهبعذابهم في الليـل أو بنهـار ؟مرقوا من الدين الحنيف لأنهـممـا وحّـدوه بـذلّـةٍ ووقــارماذا أقول وقـد تبيـن فسقهـمووعى تبجّحَهم أولـو الأبصـارولسوف يلقى الظالمون جزائهـمويصيبهـم بمـعـرّةٍ وصـغـارويصيبهـم رب العبـاد بنقمـةٍفالله لا يرضـى علـى الكـفّـارأو قد نسوا سعداً وكيف أذاقهـممرّ الهزيمة أو نسوا ذي قـار ؟أم قد نسوا القعقاع كيف أذلهم ؟فالقادسيـة مـوعـد الأحــرارستكون إحدى الحسنيين فمرحبـاًبالنصر أو نحظى بخيـر جـوارهم يبغضون المسلميـن لأنهـممحقوا جيوش الفرس كالإعصارتالله لن يأبى الشهـادة مؤمـنٌوسنخـرس الأنجـاس بالبتّـار
