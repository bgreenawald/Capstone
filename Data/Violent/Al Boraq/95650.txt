(1) لو عرفتم الله مثلي لما ترددتم حيـــــــــــــــــارى... أن تجيبوه بقربى مهما آلمتكم مــــــــــــــرارا... اشهد اللــه العظيم أني اشتريت جنـــــــــــــــــانه... بدنيا هي حقيرة ولم اقوي على عصيـــــــــــانه... عندما حاول أعداءه صدي عن سبيــــــــــــل جنانه... بزخرف المال و الدنيا وما استطـــــــــــــــاعوا إتيانه... فإذا بربي الحليم يأبى علي هجرانــــــــــــــــــــــه... فغدوت التمس نوره كلما حاولوا حجب الشمس عني... فإذا بنوره يضيء قلبي فغدوت اسطع نورا من حولي... وتحسست آياته فإذا بها اقرب إلي من قلبـــــــــــــي... فرأيت بها فارسا مقداما للواء الدين رافعا لم يولــــــي... فعاشت رؤياه في قلبي وكأني أناجيه و يسمعنـــــــي... والتمس الصمود كلما رايته يغسل لحيته ليقوم يصلــي... ويتركها عفنه في مكانها أحقر من الحيوان في التدني... وكأني استرق السمع إليه وهو يقرا خواتيم &quot;البقـرة&quot;ّ... وأقول &quot;ربي,,لما حرمتنيه قد طـــــال صبري و نفد&quot;... فيا من حجبت قلبي عن الضلال و جعلته في الأرض وتــــــدا... أشكو إليك حال القلب قد زاد به الأنيـــــــــــــــــــــــــن و اتقد... يصبرن لان حاله من أمرك فإما الرضا به أو الإعراض و السخط... ومالي إلا رضاك ربي فهلا أذنت اله الكون من لدنك فرجـــــاً...
