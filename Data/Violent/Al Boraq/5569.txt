سليمان العلوان؛ الله أكرمه فأصبح آية حبّر قصيدك واجن من ثمراتـــهِ ++ وانظم حلال السحر من أبياتهِ ترجم معاني الشكر للشيخ الذي ++ بزّ الشيوخ وفاق كل لداتـــهِ شيخي سليمان بن علوان إلـــى ++ درب الهدى يهدي مريد نجاتهِ الله أكرمــــــه فأصبح آيـــــــة ++ في الحفظ والإتقان من آياتهِ يا شيخنا اثبت فالحيــاة قصيرة ++ والعمر يحرق مسرعا ورقاتهِ لكم المحبة في القلوب أصولهــا ++ رسخت تميز أصلها بثباتـــهِ سارت بذكرك في البلاد ركائب الـ ++ ـتوحيد إنك من إجل دعاتهِ أحييت من علم الحديث أصولــــه ++ عمن تقدم من كبار رواتـــهِ وأخذت بالأمر القديــــم وحبــــذا ++ قدم المبادىء دون مبتدعاتهِ وبذلت نفسك للجهاد مجاهـــــدا ++ والله إنك في ذرى قاداتــهِ وهجرت أبواب الملوك بعــــــزة ++ فحباك مالكهم جزيل هباتهِ هب أنهم منعوك هل في وسعهم ++ أن يمنعوا عن قلبكم لذاتهِ؟ أو يمنعوا مدد الإلـــــه وفضلــــه ++ عنكم وجزل عطائه وصلاتهِ؟ ففداك كل منافـــــق ومداهــــــن ++ العلم سلمه إلى شهواتـــهِ إبليــــس جنّده لخدمتـــــه متـــى ++ ما احتاج صوتا كان من أصواتهِ خلت الديار فما بهـــــن مناصــــر ++ للدين منتصب لدحر عداتـــهِ إلا بقيــــة ثــلــــة منصــــــــورة ++ قد زادها الرحمن من رحماتهِ سلكوا على نهج عليـــه أئمـــة الـ ++ إسلام من ساداته وثقاتهِ هذامن الشعر الذي قد كان في الـ ++ إمكان فاقبله على علاتهِ الله يرفع قدركم في هذه الـ ++ دنيا ويوم الدين في جناتـــــه منقول من شبكة مهاجرونِ
