بسم الله الرحمن الرحيم )فَرِحِينَ بِمَا آتَاهُمُ اللَّهُ مِنْ فَضْلِهِ وَيَسْتَبْشِرُونَ بِالَّذِينَ لَمْ يَلْحَقُوا بِهِمْ مِنْ خَلْفِهِمْ أَلَّا خَوْفٌ عَلَيْهِمْ وَلا هُمْ يَحْزَنُونَ) (آل عمران:170) يسر الهيئة الاعلامية في جيش الفاتحين ان تقدم لكم قصيدة في رثاء ( ابو عبد الرحمن) امير قاطع الرمادي لفضيلة الشيخ ( ابو عامر العراقي) عضو الهيئة الشرعية في جيش الفاتحين في رثاء ابي عبد الرحمنأسد البواديهم أكسبوك من السباق رهانـــا ...... فربحت أنت وأدركـــوا الخسرانا هم أوصلوك إلى مُناك بغدرهم ...... فأذقتهم فـــوق الهــــــــوان هوانا إني لأرجو أن تكــــون بنارهم ...... لما رمـــوك بها ، بلغـــــت جِنانا غدروا ببسمتك الكريمةِ جهـــرةً ...... أبشـــر فـقـــد أورثتهم خــــــذلانا أهل الإساءة هم ، ولكن مادروا ...... كم قدمـــــوا لشموخـــك الإحسانا لقب الشهادة مَطْمَحٌ لم تدَّخِـــــرْ ...... وُسْعَــاً لتحمله فكـنــــت وكـــــانايا أيها المغوار ، كـنت مفوهـاً ...... بالصمت ، كان الصمت منك بيانا قد كنت في الأهوال مِسعَرُ حربِها ...... وشموخُ صبرٍ أعجــــــز العدوانا فرحي بنيل مُناك يمزج دمعتي ...... ببشارتي ويخفِّـــف الأحــــــزانا وثَّقْتَ بــالله اتصالــــــك حينما ...... صليت فجرك تطـلــــب الغفرانا وتلوت آيــــــاتِ الكتابِ مُرتِلاً ...... مُتـــأمِــلاً .. تــتــدبرُ القــــــرآنا ووضعت جبهتك الكريمة ساجدا ...... إن السجــــــــود ليرفع الإنسانا وخرجت يتبعك الأحبة ، مادروا ...... أن الفـــــــراق من الأحبة حانا معك استلذَّ الموتُ ، صار وفاؤه ...... مثلاً ، وصــــــــار إباؤه عنوانا إني لتسألني العدالـــــةُ بعدمـــا ...... لقيَتْ جحود القوم ، والنكـــــرانا هل أبصرتْ أجفانُ أمريكا اللظى ...... أم أنــهـــــــا لاتملك الأجفانا ؟ هل أبصروا أسداً يركز نظرةً...... لما تـنــاثــــر في الصباح عيانا أين الحضارةُ أيها الغرب الذي ...... جعل الحضارة جمرةً ، ودخــانا عذراً ، فما هذا سؤالُ تعطُّـفٍ ...... قد ضلَّ مــن يستعطف الـبــركانا هذا سؤالٌ لايجيـدُ جوابــــَــــه ...... من يعبد الأَهــــــــواء والشيطانا***إيهٍ أخا الدكتورِ ، إن ودَّعتـنا ...... فلقد تركـــت الصــــدق والإيمانا أنا إن بكيتُ فإنما أبكـي علـى ...... مليارِنا لمـا غــــدوا قُطعـــــــــانا أبكي على هــذا الشتاتِ لأمتي ...... أبـكـي الخلاف الُمرَّ ، والأضغانا أبكي ولي أمـــلٌ كبيرٌ أن أرى ...... في أمتي من يكســــــــر الأوثانايا فارس الصهواتِ وجهُكَ لم يكن ...... إلا ربيعاً بالهــــــــــدى مُزدانا في بِِشرِ طلعتِك الكريمةِ صورةٌ ...... للفجــــــــر حيــن يبشِّر الأكوانا فرِحَتْ بك الحورُ الحسانُ كأنني ...... بك عندهـــــن مغرِّداً جَـــذْلانا قدَّمْتَ في الدنيا المهورَ وربما ...... بشموخ صبرك قــد عقدت قِرانا هذا رجائي يا أبا ( عبد ) الذي ...... شيَّــــدتُ فــــــي قلبي لـــه بنيانا دمُك الزَّكيُّ هو الينابيع التي ...... تسقي الجذور وتنعش الأغصانا روَّيتَ
