زفــــرات ســـجـيــــــن هذه القصيد ة خرجت من رحم سجون الطاغوت الموريتاني حيث يقبع اخوانكم فيها ونرجومن كل المجاهدين نشر هذه القصيد والدعاء لصاحبها بأن يفرج الله كربه ويفك أسره وهو للعلم أحد أكبر طلبة العلم في بلاد شنقيط في هذه الايام وهو سلفي جهادي يقارع الطواغيت بلسانة وقلمه 9 شعبان 1427 محمد سالم المجلسي الشنقيطي خليلي لاتذر ي الدموع الذوارفــــــــا علي ولاتبعث إلى الخوالفــــــــــــــــــا فلا زال يسقي وابل العز موقـــفـــــي ولا زال صبري ناعم الغصن وارفــــا وكم خفـت من كيد بدا لي بـــرقــــــه فأ مـــنــني الرحمن ماكــنت خائفــفـــا فلا السجن إن راغـت ذ ئاب ظلامــه إلى نفــسي يملي علي المواقـــفـــــــــــا ولا الـبيـن إن بان الصديـق وملــــني وعـــكــر أيام الصفــا ء السوالـــفـــــــا تروضت في روض الــفـضيلة يافعا أعا طي كـؤس الود سمحا ملا طفــــــا وحرض وقع الحق قلبي فسماح بي بــساح يجا في من يعاف المثا قــفـــــــا لنا سلف عاشوا خطوبا تــتا بعــت إذا أبصرتها الشهب تبقي رواجــفـــــــا فـصف لبني الاسلام سيرة جـمعـنا فــفــيها يطيب الوصف إنكنت واصفــا لــبســنا من الأخلاق أحسن حلـــــة تميت الهوي شوقا وتــســبي المتاحــفــا وكم مسجد تاقــت إلــينا رحا بـــــه نـلـث به نور الهـدي والمعــا رفـــــــــا وكم ضمنا ربع جفا الأنـس أهـلــــه فعاد إليه الطير وازدان غـــــا ضفـــا ولاح جـــلاء الحــق منا لـفـتـيـــــة فــداســـو برجل العــز ماكـا نــزائــفـــا وأســلا فـنا كـا نو بذا الد ين أنجمـا وكـا نت عــواليــهــم بــروقـا خـواطــفــا وكــا نـو إذا دوي صهيل خـيولهـم تــهــب مــطا ياهــم سـيـولا جـوارفــــــا ســنـبقي كماكـا نو وفاء ا لد يـنـنـا نحكم فــي كــل الرقــا ب المصا حـفــــــا
