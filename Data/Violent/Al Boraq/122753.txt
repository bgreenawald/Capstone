من لا يجوز قتله في الجهاد اتّفق الفقهاء على أنّه لا يجوز في الجهاد قتل النّساء ، والصّبيان ، والمجانين ، لما روي عن ابن عمر رضي الله عنه : (أنّ امرأة وجدت في بعض مغازي رسول اللّه صلى الله عليه وسلم مقتولة ، فنهى عن قتل النّساء والصّبيان) رواه مسلم . وكذلك لا يجوز قتل الشّيوخ عند جمهور الفقهاء ، وبه قال مجاهد ، لما روي أنّ النّبيّ صلى الله عليه وسلم قال : ( لا تقتلن ذرية ولا عسيفاً ) رواه بن ماجه ، ولما روي عن ابن عبّاس في قوله تعالى :{ولا تَعْتَدُوا} يقول :' لا تقتلوا النّساء والصّبيان ، والشّيخ الكبير ' وروي مثله عن أبي بكر وعمر رضي الله عنهما. ولأنّه ليس من أهل القتال فلا يقتل كالمرأة ، وقد أومأ النّبيّ صلى الله عليه وسلم إلى هذه العلّة في المرأة الّتي وجدت مقتولة في بعض مغازيه ، ( فقال : ما كانت هذه لتقاتل ) رواه داود . لكن السؤال هو ما قول الاخوة بالنساء والاطفال ومن لا يجوز قتلهم في الجهاد عندما يقتلون في العمليات الجهادية؟
