. اذا جـــــــــــانــــي الــخـيــــــآآل ... بِ طــــييــفـــك يَ الــجــمــــــآآل اعــيـــد الـــذكـــرى يـــمّـــهـ والــــدمـــع سـيّــــآآل مــشـيــتـــــي يَ الــحـنــــــآآن ... وتـــــــعّـــــبــــنـــــــي الــــزمــــآآن ابـــي هـــمــــســــك يـقــول تـنــام ويــا مـا كــآآن بــس طــلّـــي شـــوي هـِنــــآآ ... يـــمّــهــ طــيــفــــك لــي هـَنـــــآآ مــشـتـهــــي لِ رووحـك آنـــــآآ ... كـــل شـــي بـــعــــدِك فـــنــــــآآ آنــا يمّـهـ آعيش الدنيا رووحي ... مــــحـــدٍ ضـــمّــد جـــروووحـــــي يـــــتــــيــــــمٍ مـــــــالـــــــهـ آي صـــــــوووووووت يـــــتــــيــــــمٍ مـــــــالـــــــهـ آي صـــــــوووووووت يـــــتــــيــــــمٍ مـــــــالـــــــهـ آي صـــــــوووووووت يـــــتــــيــــــمٍ مـــــــالـــــــهـ آي صـــــــوووووووت نششيد آلييم بِ صووت طفل لتحميل النشيد اضغط هنـــــــــــــــــــــــاااااااا الله يححفظ لنآآ أمهاتنـآآ
