السلام عليكم... أحبتي في الله إن الموضوع جد خطير ... والأمر لا يحتمل التأخير اقرؤوا معي هذه القصيدة من الشيخ حامد العلي وبتمعن : هي الروافـض لاعجـــبٌ تعادينـــــــــــي فالودُّ منهــم بعيـــــــــــــــدٌ لــــــنْ يدانيـــني قد طال ضحْكي من التهديد مندهشا وغطّني الضحْك حتى كاد يبكيني أبشر بأمْن يامـــــــــــــــــــن بات يُوعـــــــــدُه مطيـــــــّـة الكفـــــــــرِ أذنابُ الشياطين قد كان بالله لــــــي حصــــــنٌ وملتجأ فحسبــــــــي الله يحفظنـــــي ويحميـــــــــــني وأسأل الله معتصمــــــــــا بعزّتـــــــــــــــــــــه نصــــــــــــــرا ًعزيزاً وبالإيمــــان يحييــــــني ويـــــــــــلٌ لهـــــــــــــم ممـــا ســـــــــوف تفعلـــه كتائبٌ الذبــــحِ مـــن حزّ السكاكين صنائع المجـــــــدٍ مـــــن أبنـــــــــــــــاء (قاعــدةٍ ) وراية ُالعزّ تسمـــــُـــــــو، شـــــمَّ العرانين ما مات سيّدهم بل جـــــــــدّ وارثـــــــُــــه يقفـــــــــــــــو صنائعـــــــــه فعـــــــــــلَ الشواهينِ ما عابني غــــير أنــــــّــــي اليـــــــــوم منتصـــر ٌ لذلـــك الليـــــث ذبــّــــاح الملاعيـــــــــــــــــنِ ما كنتٌ أحســــــب (بغدادا) تودّعــــُـــه إلاّ بدمـــــــــــــع جرى مـــــــن قلــب محزونِ كماء دجلة يجـــــــــــــــري ليس منقطعا لعله إنْ جـــــرى بالدمـــــــــْـــــــــــــع يُسليني عمــــــــــــــــــر الشهيد سيخلد بعد قاتلــه وبالشهيــــــــــــــــد حيــــــــــاةُ الحقّ والديــنِ أما اللئيم فميـــــْـــــــــت قبـــــــل ميتتــــــــــــه وإن جاوز العمــــــــــرُ تسعا بعــــــد تسعينِ حامد بن عبدالله العلي عرفتم ما أقصد ..!!؟ قد طال ضحْكي من التهديد مندهشا وغطّني الضحْك حتى كاد يبكيني الشيخ حامد يتعرض للتهديد من قبل الروافض الحاقدين أبناء المتعة ... وللأسف الشيخ حامد عالم ضاع حقه بين أمته حاله حال المجاهدين منسيين ومتروكين .. ولكن &quot; كلا إن معي ربي سيهدين &quot; . . . فهذه رسالة أوجهها إلى كل من يعرف الشيخ ، وإلى كل من هو قريب من الشيخ بأن يعدوا أنفسهم ويتجهزوا ويحموا شيخهم وهو شيخنا وعالمهم وهو عالمنا .. . . أين أبطال الكويت ..!!
