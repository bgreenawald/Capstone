من شريط يا امتنا .. نشيد صبحيهم للمنشد المجاهد أبو أسيد المدني صبحيهم صبحيهم أيقظيهم واحكمي الإثخان فيهم زلزليهم يا كتائب وانفضيهم كالرمــــــال يا أسود الله سيروا واضربوا أعلى مثال في قتال الكفر هيا واهزموا أهل الضلال اســــتعـــيـدي المـــــــجــد فــيـــــنا ـــــ فـــزمـــــــان الــــــــذل طــــــــــــال واجـــــــــعلي الــتــــــاريخ يشــــدو ــــ ببــــطــــــــولات الـــــــــــــــــرجال حطمي قيد أسير ذاق ألـوان النـكـــــــــــــــال لا تصمي السمع عنهم مثل أشباه الرجــــــال أشعلي الأرض سعيرا تحـت أقــدام الحـثـــال جلجلي التكبير فيهم واســرجـي خيـل النضال إن نصر الله آت فاصبري عند النزال ـــــ ذكرينا بزمان فيه كنا كالجبــال بزمان ليس حلما ليس طيفا من خيال ــــ فيه سعد والمثنى وأسـيد وبلال فيه عَمروٌ ومعاذ فيــه ســيف اللــــه صـال أمة التوحيد قامت وكيـــــــــان الشرك زال وأعز الله جندا حكموا الدين فعـــــــــــــــال جعلوا الحق كنانا وغدوا فيه نبــــــــــــــال للتحميل ( برابط مباشر ) صبحيهم ولا تنسوني من صالح الدعاء أخوكم أبو المجد
