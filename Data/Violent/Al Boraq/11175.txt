مــالــي والـعـبــاد شــــر عــبـــادمـالــي والـفـسـاد قـــد عـــم الـبــلادمــالــي والــنــاس فـــــي غـــــيوضـــــلال ومـــجـــون وإلـــحـــادتبا لكم يا قوم تبا على فسق ولهـو وعبـادة أوثـانأمــــا الـصـالـحـون فــــلا تــســـلفـهــم فـــي عـــداد الجـبـنـاء الـنـيـامماذا أقول والقلب علا همـه والعيـن تبـث الأحـزانمللنا وسئمنـا ضياعـا شبابـا فـي غيـه سكـرانفقفوا أمتي نبكي سويا على الذل والضعـف والخـذلانأمة غرقت في بحر ظلم وعصت محمـد سيـد الأنـامقولوا بربكم ما أصابكـم أو اصمتـوا لحيـاة الهـوانوالـغـوانــي عـنـهــم فــــلا تــســلفـالـكـلام لابـلـيـس والـحـركـات لـجــانوالربا طريق كل بيع وديـن وقـرض وحتـى الكـلاموالبغض والكره والحسد حال الأقارب والجيران والخلانأما الأقارب فلا تقارب والأباعد فخير لك مـن الأعمـامالكـل شغلـه شغـل ليـس إلا شغـل زائـل فــانوعقوق الوالدين فشر فساد وقع وصار بلية الأوطـانسحر وشعوذة وتلبيس على ملك الملوك الملك الديـانمالي والنساء المائلات يملن في غياب دين وقلة إيمانإيمان ماذا والرقص قد حل مكـان الصـلاة والصيـامأف لكم ولمـا خنغتـم لدينـا ومـال وحـب سلطـانحب وهيام وضياع عقل كيف يكون لغير الواحد المنانونظرات إبليس تأتي به فحقده قد علا وفـاق العنـانأقسم بعزة الإله أنـه سيغـوي ويضـل كـل الثقـلانلكنه خاب وخسأ لأنه لن يستطيع لأهل االجهاد والإيمانهذا حال أمتنا بل نقطة مـن بحـر ملـيء بالعصيـانيا قوم ارجعوا وتوبوا إلـى رب غفـور كريـم حنـانغافر الذنب وقابل التوب شديد العقاب لأهل العصيـانيا رب تب علينا وعلى كـل مذنـب يرجـو الرحمـنوالصلاة والسلام على محمد خير البرية عليه السـلامصلى عليك الإله وسلم حتى الملائكة الأبرار والسـلام
