بسم الله الرحمن الرحيم الحمدُ للهِ والصلاةُ والسـلام على رسـولِ الله وإلهِ ومن واله وبعدُ : أزفُ إليـكم بـشرى إستشهـاد أحد أسـود بغداد بعد علميةٍ طيبة ومبـاركة لينـال بعدهـا الشهـادةَ مقبلاً لامدبراً بإذن الله تعـالى تقبلـك الله يــاأبــاعبدالله تقبلـك الله يــاأبــاعبدالله تقبلـك الله يــاأبــاعبدالله تفــاصيلُ العـملــية هـاجم ثلاثةً من شُجعـانِ وفـرسـانِ بغداد - قـائدِ صحوة التـاجي - الكلب المدعوا - شذر- فيـسر الله لهـم تنفيـذ حُكمِ الله عليهِ ألى وهـو قـتـلُهُ وبالإنسحـاب قـدر الله ومـاشـاء فعل إن السـيارة التي كانوا يستقلونهـا قـد عـُرفت وتـم الإبلاغُ عنهـا في السيطرات التـابعة لصعـاليك صحوةِ إبليس فحـاولت إحدى السيطرات إعتـراضهم لإيقـافهم ولكن خسئوا وخابوا فهؤلاءِ أُسـودُ التـوحيد وكلُ الفخـر فقـام الإخوة بــالهجـوم عليـهم بالأسلحةِ الخفيفة ليقتلوا بعضً منـهم وقـدر الله أن ينـال الشهـادة أخينـا الحبيب - أبـا عبدالله - الشهـادة بإذن الله تعالى مقبلاً لامدبراً ضـاربـاً على عدوهِ بنيـرانِ سلاحهِ البسيط ليهرب الإخوة الأثنين إلى أحدى القـُرى القريبة ولكن قـدر الله أن تنفذ ذخيرتهم فتم أسرهم من قبلِ صحوة إبليس فحـاولوا التحقيق معهـم لإخـذِ بعضِ المعلومات التي ربمـا تنـفعهم بشيءً ولاكن كـان الإخوة وللهِ الحمدُ بقمةِ الصبـر والثبـات بفضلِ اللهِ ومنـتهِ عليـهم فمـا نطقوا حرفـاً واحداً فإستفزوا صحوة إبليس ليقوموا بإعدامهم والله المستعان بالـرصـاص فتقبلهم الله وأعلى نزلهم عندهُ سيـرة أخينـا الحبيب أبــا عبدالله .. [ شـُجــاعُ الــكرخ - غربِ بغداد ] إلتحق أخينـا الحبيب أبـاعبدالله بالمجـاهدين بعد دخولِ قواتِ الصليب بوقتِ قصير فأخذ موقعهُ مع أخيهِ وصحبهِ في المعـركة فكـانوا أشداء على الكفـار أثخنوا بـهم وللهِ الحـمدُ , كـانت لأخينـا أبـا عبدالله تـوجهـاتٌ - قريبة من الـصوفية - ولـكن وللهِ الحمدُ بعـد سنةٍ من العملِ مع أخوةِ العقيدة الصحوة - نحسبهم ولانزكي على اللهِ أحداً - طـهر الله قلبـهُ وفتـحَ عليـهِ فتحـاً مبينـا , فعمل وعمل ضد أعداءِ الله من الـرافضة بشكلٍ كبيـر بل كـان أشـدَ الإخوة عليـهم وكـان من المحبين لشيخنـا أبي مصعبٍ عليـهِ رحمة ُ الله وكـان كثـيراً مـايتأثرُ بكلامهِ ولايستأنسُ إلى بـهِ في جلسـاتهِ كـان حزينـاً مهمومـاً على حـالِ الجهـاد والمجـاهدين وكـان نصـوحـاً شفوقـاً محبً ومحبـوبـاً لدى الجميع فبـعد أن دخلت خطةُ فرضِ القـانون كـان لهـا بـالمـرصـادِ في ديـارهِ فبفضلِ الله أفشل سيطرة القوات الرافضية على المنطقة هـو ومن معهُ من الإخوة وبعد فترةٍ أصبح المطلوب الأول لقـواتِ الصليب في منطقتهِ فخـرج منهـا مع رفيقِ دربهِ ..[ زيـد ] رحمهُ الله وأستمر جهـادهم وإجتهـادهم وقـد عـانوا كثـيـراً من الفتن التـي جـابت أرض الجهـاد والشبهـات الكـثيـرة ولكن الله إصطفـاهم وحفظهم بحفظهِ سبحـانـهُ وتعـالى , فبعد مدةٍ ليست بالـطويلة رحل رفيقُ دربهِ وأخيـهِ الذي لم تلدهُ أمـهُ بعمليةٍ مبـاركة ضـد الـحرس الوثـني فقـد إنقض رحمهُ الله على ثلاثةٍ من الحـرس الـوثني بسلاحهِ الخفيف - المسـدس - فقط ليمكنهُ الله من قتل إثنين منـهم ولكن قـدر الله إن الثـالث تمكن منـهُ لينال من اللهِ ماأحب نحسبهُ كذلك ولانزكي على اللهِ أحداً , وبعـدهـا أستمـر جهـادُ أخينـا أبو عبدالله رغـم إن بعض الحُزنِ أصـابهُ على فقدانِ رفيقِ دربـهِ ولـكن لـم يثنيهُ شيئـاً عن طـريقهِ وميسـرتهِ فـخرج أخيـراً رحمهُ الله تعالى إلى عمليـتهِ الأخيـرة مستبشـراً فقـد كلمنـي رحمهُ الله قـائلاً قبل التنفيذِ بسويعـاتٍ : [ شـلونك حبيبي ..[ فلان ] أن شـاء الله بـخيـر أكلك أشوفك منـا لكم يـوم بس أجي بس أدعيلي أكوا شغلة زينـة راح تصـير بإذن الله بس إدعي للإخوة ] فقلتُ لهُ : [ أنتــه إلي راح تسـوي هـاي الشغلة ] فقـال لي ضـاحكـاً رحمهُ الله فكـان أكثر النـاسِ تواضعـاً للهِ ولعبـادهِ : [ يمعـود هـو أني مـال هيـج شغلات جبيـرة ] فبـعد هذه ِ المكـالة التـي أرعبتنـي وأفـرحتني .. !! صبـاحــاً رررررررررن الجــوال فأجبتُ فقيل لي [ السـلامُ عليكم شلونك أخوية] . [ تـرى اليــوم توفى أخــوك أبــو عبدالله ] آآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآه ثـم آه فوالله الذي لاإله إلى هـو مـاوجدتُ إلى كلمـاتِ أبي مصعبٍ رحمهُ الله وهـو ينعى ويـرثي أبي أنسٍ رحمهُ الله عينُ جـودي بدمعكِ الرقراقِ وأسكبيهِ على أعز الرجالِ أيُ خطبٍ قـد أثـار شجوني وأشـاع الأحزان في أعمـاقي يـاأخي يـاأخ المـودةِ والحُبِ فخذهـا من قلبي الخفاقِ آآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآآه رحمك الله أبــا عبدالله فدمُك نــور ونــــار أبــا عبدالله وأبي العينــاء أهٍ يــالهُ من يومٍ زرتُ فيـهِ أبـا عبدالله في دارهِ فأخذتُ معي ( الفلاش رام ) الذي جمعتُ فيهِ الإصدارت الجهادية ليُشاهدها أخي أبـا عبدالله فكـان من بينهـا إصدارُ دولـةِ العز - غزوة الدكتور فتحي - فبعد وصولي إليـهِ شغلنـا الفلاش سويةٍ فوصلنـا إلى غزوة الدكتور فتحي فبدءت الشـريط وبدء أبـوعبدالله بذرف دموعهِ وضحكاتهِ وإستبشـارهِ برؤية الإخوة الإستشهادين رحمهم الله وإيـاه وتقبلهم عندهُ فوصل بنـا المطـافُ لأبـي العينـاء المهـاجر فذرف دموعـاً لم يذرفهـا قط ُ من قبل فبكـا بكاءً الفجيع بإهلهِ وذويهِ وكـان يصيح بإعالي صوتهِ اللهم إجمعني مع أبي العينـاء اللهم إلحقني بإبي العينـاء اللهم أجعلني كأبي العينـاء آآآآآآآآآآآآآآه يـاأخوة لـو تعلمون كم أثر بهِ أبــا العينـاء فللهِ درُ أبي العينـاء ورحم الله شهدائنـا مهـاجرين وأنصـار وألحقنـا اللهم بهـم ثـابتين صـابرين على ماأصـابنـا وصلى الله على محمدً وإلهِ وصبحهِ وسـلم أخيكم محارب لاتسنون إخوانكم المجاهدين من الدعـاء
