(مدينة الأفراح)1-ماذا أقول وقد تراخت هــــمتــــي سقط الشراع وغاص بالألواحِ 2-ماذا أقول وحالُ مَن مِن حالنــــــا دخل التتار مدينة الأفـــــــراحٍِ 3-دخل التتار وكلهم حقدٌ لنـــــــــــا حقدٌ دفينٌ مـــــــذّ محمــدِ لاح 4-هدموا البيوت وخرّبوا أوطاننــــا كبتوا الحياة وشعلة المصبــاح 5-قتلوا الشيوخ وهتّكوا أعراضنـــا وتناوشوا طفلا مع الأقـــــداح 6-إن الكلاب تكاتفت وتحالفـــــــت للنيل مِمْن قام بالإصــــــلاح 7-حتى الخنازير اللعينة وافقــــــت.. للحلف والثعلب لهم قـــــــدّاح 8-أين الشعوب ألا يرون دمائنـــــا .. تجري ويشربها بنو السفّــــاح 9-أين الحماة عن البلاد وأمنِــــــها.. باتت ضمائرهم بـــــلا أرواح 10-أين الجيوش جيوشنا في غفلــــةٍ وإجازةٍ طالت مع السّــــوّاح 11-أين المداوي كي يعالج ضعفنـــا نُفيَ الطبيبُ وزُجَّ بالجــرَّاح 12-ضَعُفَ الشبابُ ورُمَّ جُلَّ سلاحنا ظلّت ذخائرنا بلا مفتــــــاح 13-لبيك يا أرض البطولة والفـــدى فدماءنا تفديك كـــــل رواح 14-لبيك بغداد العروبة دائــــــــما فالنصر أقبل مشرق الإصباح 15-لبيك لا تهني ولا تستسلمــــــي واستمسكي بالدين خير سـلاح 16-آن الأوان لسلخ أذناب العــــدى أو قطفُ رأس الغاشم النبـــاح 17-الشعر ضربٌ للجهاد وصولــــةٌ إن مال للحق بكل نجــــــــاح 18-كل من أجاد الشعر حقا قالــــــه فمتونه لا تحتمل أمـــــــــلاح 19-يأتيك إن صدق الفؤاد لعقلـــــه كالغيث يجري سيله نطـــــاح
