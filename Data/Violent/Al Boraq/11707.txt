السلام عليكم ورحمة الله وبركاته وا كــــــــــــــرباه وا كــــــــــــــــــــــرباه وا كــــــــــــــــــرباه اخواني في الله انني اخوكم في الله ارجوكم ان لا تتركوني فوالله الذي لا اله الا هو انني في هذا اليوم فجر الاربعاء حل بي كرب عظيم واغلقت جميع الابواب في وجهي ولم يبقى لي سوى باب رب العالمين اخواني ادعوا لي بان يفرج الله كربي وهمي فوالله انني في غم وهم وشدة واسأل الله ان يثبتني ثبات الابطال وان يزلزل الارض تحت من ضيق علي في هذا الفجر وفي كثير من الايام اخواني والله انني صابر محتسب اخواني اكثروا الدعاء لي فوالله ان الدعاء لاخوكم المسلم في ظاهر الغيب مستجاب ادعوا ان الله يذهب عني هذا الحزن والكرب والهم ولا اقول الا حسبي الله لا اله الا هو عليه توكلت وهو رب العرش العظيم اللهم احرسني واحفظني واذهب همي واذهب عني هذا الكرب
