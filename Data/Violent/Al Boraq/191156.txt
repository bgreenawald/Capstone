هذه قصيده للشاعر الدكتور عبد الرحمن العشماوي في أمه وهي جديد العشماوي أمـــــاه أمــــاه غيث المقلة انسـكـبــا في يوم عيدي وجمر الحسـرة التهبـا قالوا : أتى العيد بالأفراح ، قلت لهم : بـالـعـيـد نــفـرح لــكـن الأسى غلبـــا لـمــا أتـى العـيد بشـرت الفــؤاد بـــه فاستبشـر القـلــب لكــن الأنـيـــن أبـى يـا ويـح نفسـي التـي ما بين فرحتهـا وبؤسها ترقب النـور الذي احـتـجـبـــا نــبــادل الـنــاس يــا أمـــاه تــهنئـــةً وكـل معـنـــى للفـــظ الفرحـة انـقـلـبــا أهــلاً وسهــلاً لـمـن يــأتي نـرددهـا ومــارد الحـــزن فـي أعـمـاقـنــا وثبـا وكـيف تسـلم لـي فـي الـعـيد تـهنـئـة ولم أكــل بـنـــور البسـمـــة الــهـدبـــا ولــم أقــابـلـك يــا أمــاه فــي سـحــرٍ قبل الـصلاة لأنســـى عنـدك النصبـا ولــم أجـدك على الكـرسي جــالـسةً كالبدر ما غاب عن عيني ولا احتجبا ولــم أقـبـل جـبـيـنـا طــاهـرًا وأرى وجهًا حبيبًا يرينـي البــدر و الشهبـــا أواه أواه يـــــا أمـــــاه مــــن ألـــــم مازلـت أكتمـه فـي خـاطـري لــهبـــا يــكــاد يـقـتـلـنـي شـوقي الكـبـير إلى سماع صوتك ، صوتـًا لـحنـه عذبــا لـمــا تنــادين باسمـي يزدهي حـلــم به أرى كـل مـا ينـــأى قــد اقـتربـــا بيــن الـمصلى ومستشفاك كنت على مشــارف الأمــل الميمــون مـرتقـبـــا تزاحمت صـور الذكرى على خلـدي طـفـولـةً وشبابـــًـا بــاذخــًـا وصــبــا وتضحيــات رأينــاها رسمـت بــهــا طريقنـا ومســحــت الـهــم و التعبـــا تــدور فــي فـلـك الأيـــام ذاكــرتـــي تعيد من صـــور الأحــلام مـا ذهبـا رأيــت أمــي إلــى الــوادي مـهرولةً لتجلــب التيــن للأطفـــال و العـنبـــا رأيـتــها في طريــق الـبئر غـــاديــةً تجـــر بالـحبــل دلـــوًا تمــلأ القربــا رأيتـهـا فـي حـمــى ظبيــان بــاحثـةً عن يابس الطلح حتى تجلب الحطبــا رأيتها إنــهــا أمــي التـي حــمــلـــت عنـــاءنــا فغــدت أمـــًـا لنـــا وأبــــا أمــاه دونـــك صــار البيـت مــلتحفًـا بـحزنـــه وأتــانـــا الـعيــد مـكتئبـــا يا شـمس منزلـنــا لا زلــت ساطعــةً تـمــزقـيــن دجــى آلامــنـــا إربــــا إنـا طلبنـــاك مـمــن ليس يعــجـــزه شـــيءُ ونسـألــه أن ينجــز الطلبــا ما استجـلـب الـمـرء شيئًا بالدعاء له إلا وكان الرضـا من بين ما جـلبــا لـولا ســكـينـة إيمـــان بخـــالــقــنــا بهـــا نسكّـن وجــدانًـا إذا انتــحبـــا لأبـصر النـاس من آثــــار حسرتنـا شـيئًا عظيمــًا ومن أحزاننـا عجبـا يقيننــا مــكّـن الصبر الجــميــل لـنـا ومـــد للقلـب نحـو الخالـق السبـبــا فـنـحن نطـمـع فـي إنـعــام خــالقنــا ونرتضي ما قضى فينـا ومـا كتبـا أمـاه صبـرك فــي الــدنيا غــدا مثلاً فما رأيناه فـي درب الجـراح كبــا كـذلك الأم يـــلقـى قــلبـهــــا عـنـتــًا مـن الحياة ولـولا الصبر لانشعبـا مـــا الأم إلا ينـابيــع الحنــان جــرى معين وجدانها الصافي لمن شربـا ببرها تـفتــح الجنـــات ســـاحتـــهــا وتمنـح اللؤلـؤ المكنـون و الذهبــا باب إلى جنــة الفردوس يــدخــلـنــا طوبى لمن طرق الأبواب واحتسبا يزكــــو الثـرى تــحـت رجــليها فلا تربت يدا مقبّـل رجليهـا ولا تربـــا
