فوقهم بصليبــــــــــــه كثر ولكن كالغثـــــــــــــــاء تفرقت أهواءهم فجموعهم مغلوبــــــــــة جاء الصليب وجــــــــاء كل منافق كل يؤمل حظه ونصيبــــــــــــــــــه واتى المجوس وقد تـــأجج حقدهم فبليتيِ بالأنجـــــــــــــاس يا فلوجة أحفـــــــــــــــاد بن العلقمي ونسله رضعوا الخيانة بالخنــــا ممزوجة من عصر هولاكو إلي بوشٍ لـــهم في كل يوم خســـــــــــــــة معهودة وخوالف جثموا علي أنفاسنــــــــا نسجوا بسيرة إفكــــــــــهم أكذوبة ومفادها أن الجهــــــــــــــاد معطل فأطع لأمر المعتدي وعلوجــــــــه هذا التــــــــــــــآمر أمره لا ينطلي إلا علي من لبه مسلوبـــــــــــــــة جمعوا مكــــــــائدهم وقائد زحفهم إبليــــــــــــس يحشد رجله وجنوده قد أقبلوا بقرونهم وذيولـــــــــــهم وجعـــــــــــــاجع ومزامر مطبوبة وتضج أنفــــــــــــاس لهم بأزيزها حقدا يسعر نـــــــــــارها الموقودة درجت علي التضليــــل أبواق لهم تروي الوقــــــــــائع قصة مقلوبة جاءوا لنــــــــا بالسامري وعلجه وخواره أضحى لــــــــــهم أنشودة فمن الصباح إلي المساء وذكرهم هذا الإلـــــــــــــــه وذاته المعبودة ذلت رقاب المشركين ببكــــــــــــةٍ وجبــــــــــــالها قد رددت أهزوجة ولى زمــــان المشركين وشركهم والحق أسفر صبحه ببلوجــــــــــه لمــــــــــا تحدرت الجموع وهدرت بحديدها وبنـــــــــــارها المصبوبة أبت العفيفة بنت يعرب ذلــــــــــــةً وتمسكنــــــــا للشرعة المعووجة وتحصنت بالدين متراسا لهـــــــــا فتشامخت فوق السحـــاب بروجه وقفت بوجه الغاصبين عصيـــــــةً كـــالصخر .. كالجبل الأشم عنيدة أبناؤها كليوثِ غــــــــــابٍ كشرت حول الحمى تصطــــــاد كل طريدة من كل روميٍ تحدٌّ بنحــــــــــــــره أنيـــــــــــابها حتى غدت مشحوذة لما دعا داعي الجهـــــــــاد بليلها وثبت أمـــــــــام الزاحفين أسوده دوى صدى التـــــــــكبير فارتعدت للمشركين فرائص مرعوبــــــــة والله لو شهد الوقيعة خـــــــــــالد لتنفست من صدره تنهيـــــــــــــدة لا نام ضبحت بهم صبحا فأورت قدحهـا وتوسطت جمعـــــــــا فطال نحيبه زيد تقدم .. هـــــــــاك جعفر خلفه وبن الرواحة يستـــــــــهل قصيده هي عودة التــــاريخ يروي قصة للمجد في زمن النكوص عجيبــة فهنـــــــــــاك
