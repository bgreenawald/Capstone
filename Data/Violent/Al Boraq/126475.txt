بسم الله الرحمن الرحيم لكل من أحب لله وفي الله..سلام لكل من فارق الأهل والأحباب..سلام لكل من تلقي طعنة غادرة من حبيب فنزفت عيناه دماً ولم يثأر..سلام لكل من انشبت فيه الحوادث والمصائب أظفارها فصمد ولم يسقط..سلام لكل من أحبه الله فزوي عنه الدنيا وكثرت عليه الشدائد والبلوي ليقربه منه..سلام لكل من شردته قوي الظلام وباعدت بينه وبين أهله ووطنه..سلام لكل من فقد شهيداً أو قريباً أو حبيباً..سلام لكل من ترك الأهل والمال والزوجة والولد ليرابط في ثغر من ثغور الإسلام..سلام لكل من بذل روحه في سبيل الله وهو ينافح عن الدين والأرض والعرض..سلام للأمة التي كتب عليها القتل والقتال والجهد والجهاد حتى تقوم الساعة..سلام لأهل غزة الصابرة المصابرة المحتسبة..سلام عليكم سلام من الله ينزل السكينة علي القلوب ,ورحمات تفرج الشدائد عن المكروب وبركات تعم ,فلا تترك عائلاً ولا فقيراً ولا منكوب إليكم اخوتي قصيدة ســـــــــــلامي ســـــــــــــــــــــــــــــــــلامي ســــلامي حين أبعثه ينــــادي تردد صوتــــه كــل البـــوادي ويمخر من عباب الموج بحراً ليرسوا في شواطئكم يهـــادي إذا الأقدار قـد حكمـت علينـــا تشتـت جمعنــا فـي كـل وادي وينـزل مـن شديـد الأمرِ بـأسُ تلقتــه الصـدور بــلا ارتعـــادِ لنـا فـي قـدرة الرحمـن كــهف نلـوذ بـه فـلا نخشى العــوادي إلـي أهلـي أحــنُ وحيـث كنــا يجمعنــا الحنيــن إلـي بــلادي ســـــلامي حيــن أبعثــه إليـكم كمثل الصبـح يشرق من سوادِ والسلام عليكم ورحمة الله وبركاته مع تحيات أخوكم أبو المجد بالله عليكم اخوتي لا تنسوني من دعوة بظهر الغيب
