السلام عليكم ورحمة الله وبركاته الأبيات عباره عن رد على قصيده هللي واصبري .. وكان الرد على نفس القافيه (( قافية الانشوده )) دمــــــري وفـــجـــري ياصواريخ على روس اليهود حـدري الـجـهـاد هــــو كــزنــي دمــــــري وفـــجـــري متي جيوش العرب من قواعدها تنفري الـجـهـاد هــــو كــزنــي دمــــــري وفـــحـــري صفحة العز للفلوجه بالتاريح سطري الـجـهـاد هــــو كــزنــي دمــــــري وفـــجـــري وامريكا في العراق مـارح تصبـري الـجـهـادي هـــو كــزنــي دمــــــري وفـــجـــري و جيوش الكفر قامت تشرد وتدبـري الـجـهـاد هــــو كــزنــي دمــــــري وفـــجـــري والله واكبر رددي كل دقيقـه وكبـري الـجـهـاد هــــو كــزنــي دمــــــري وفـــجـــري كل قوة الارض على المجاهد ماتقدري الـجـهـاد هــــو كــزنــي ذيب الجهاد
