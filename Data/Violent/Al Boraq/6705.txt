بسم الله الرحمن الرحيم تتجلى عظمة الخالق ... في الحديث القدسي الشريف قال سبحانه وتعالى : {يا ابن آدم جـعــلتك في بطن أمك. و غـشــيت وجهك بغشاء لئلا تنفر من الرحم .. و جـعـلت وجهك إلى ظهر أمك لئلا تؤذيك رائحة الطعام .. و جـعـلت لك متكأ عن يمينك .. و متكأ عن شِمالِك .. فأما الذي عن يمينك فالكبد .. و أما الذي عن شِمالِك فالطحال .. و عـلـمـتُـكَ القيام و القعود في بطن أُمِكَ .. فـهل يقـدر على ذلك غــيري ؟؟ فلما أن تـمّت مدتـك .. أوحــيـت إلى الملك بالأرحام أن يُخرِجَكَ .. فأخرجك على ريشة مِنْ جناحه .. لا لك سن تقطع .. و لا يد تبطش .. و لا قدم تسعى .. فانبعث لك عِرقان رقيقان في صدر أُمِكَ يجريان لبناً خالِصاً .. حاراً في الشتاء .. و بارِدًا في الصيف .. و ألـقــيت محبتِكَ في قلب أبويك .. فلا يشبعان حتى تشبع .. و لا يرقُدان حتى ترقد .. فلما قوي ظهرُك و أشتد أزرُك .. بارزتـني بالمعاصي في خلواتِك .. و لم تستحي مني .. و مع هذا إن دعوتني أجـبــــــتـك .. و إن سألتني أعـطـــــــــــــيــتـك .. و إن تبت إليّ قـبــلـــــــــــــــتـك }
