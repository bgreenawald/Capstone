فترة الاصدار: 10 قد 2008, 22:56 من لي بطائرة تحلق عاليـــــــــــاً وتقلني و الصحب للشيشـــــــــــــان بعد البلاد غدا يؤرق مضجـــــعي ويصيبني بالهم والغثيــــــــــــــــــان حتى متى سأظل أرقب صابــــــراً يوم اللقاء ومرتع الفرســــــــــــــان حتى متى سأظل حملا ضائعـــــــاً والأسد قد وثبت إلى الميــــــــــــدان حتى متى يا نفس أنت كسيــــــرة تبكين مقعدة على النيــــــــــــــــران حتى متى يا صاحبي نبقى هنـــــا والروس تهدم شرعة الرحمـــــــــن ذهب الرجال بصدقهم لما مضـوا وبقيت بين الأهل والجيـــــــــــــران ذهب الهزبر الحر جوهر صابــراً وتلاه محتسباً سليم هــــــــــــــــانى ومضى السويلم والرجال تهابـــه وأبو الوليد مجندل الشجعــــــــــــان و ابن المدينة(1) مات وهو مكبــــر يهدي الكرامة كل داغستــــــــــــاني لم أنس يعقوب ابن غامد(2) ثائــــــراً يرمي الأمون وعابدي الصلبـــــــان لم انس أيضا يا أخي مجاهــــــداً القصف صنعته وكان يمــــــــــــاني(3) لله در السيف وهو مؤصـــــــــلٌ قولاً صحيحاً من هدى القــــــــــرآن قد صاحب العلماء شبلا طيعــــــاً ومضى يقود الأسد في الشيشــــــان قد نال عن شيخ القصيم(4) إجـــازة حتي يحكم شرعة الرحمـــــــــــــــن لم يشبه القوم الخوارج منهجـــاً أو كان من بين الورى فتـــــــــــــان هل تذكرون حديثه لجحافــــــــل سفكوا الدماء ودمروا أوطــــــــاني إذ قال قولته إليهم ناصحــــــــــاً لا تجرحوا حرما بني لأمـــــــــــــان بل ادفعوا الكفار عن أوطانكــــم لا تفرحوا بخروجكم شيطــــــــــــان فإلى العراق الحر هيا فانفــــروا أو كابل الثكلى أو الشيشــــــــــــــان(5) ومضى أمير الجند أصلانُ مضى باع الحياة لربه بجنـــــــــــــــــــــان وأبودجانة (6) للرحيل ركابـــــــــــه قد أسرجت ومضى بغير توانـــــــي مضى الكريم السمح وهو مجالدٌ أهل النفاق على ثرى الشيشـــــــــان(7) قد أثخن الأعداء من ضرباتــــه ومضى وحيداً بعد ذا الإثخـــــــــــان حتى متى سأظل أحصي يا أخي أهل الفضائل قادة الشيشـــــــــــــان و أنا حبيس قد جلست مكــــبلاً أبغي الخروج إلى وغي الفرســـــان يا صاحبي إني سئمت مواضعاً للسلم أسكنها هنا بأمـــــــــــــــــــان يا صاحبي أبغي الرحيل وأدمعي سالت تقرح مقلتي وجنانــــــــــــــي لم أنس يوماً شاملا ً و رحيلـــــــه كيف انبرى يرمي ذوي التيجـــــــان هذى قروزني قد بكت بمــــرارة يوم الفراق ودهرهم ألفـــــــــــــــــان(8) لم تنس فيدنو ليثنا وزئيــــــــره عند الوغى يعدو على القطعــــــــان سل عنه بوتلخ(9) يا أخي ستجيبكم عن شامل يقتص للولـــــــــــــــــدان آه وآه ثم آه يا أخــــــــــــــــــي مرض الجنان وأقعدت أركانـــــــــي لولا النفاق ونيتي يا صاحبـــي لكنت نسراً في ذرا الشيشــــــــــــان يارب فارزقني جهاداً خالصــــاً لا تعتريه مكائد الشيطــــــــــــــــــان حتى أمزق يا إلهي راجيـــــــــاً حوراً حساناً أنشئت بجـــــــــــــــنان يا رب لي صحب أحبهم وقـــــد
