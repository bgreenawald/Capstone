عسكرت أياما في بعض الأنحاء .. وهاهو الجيش يتحدَّر منصبَّاً.. يغزو على بركة الله .. وباسم الله *** سـلام الله ذاك هـو السـلام ... تعالـى ربنـا فهـو السـلام وجاؤا بالغـلام لكـي يغطـوا ... مذلتهـم بأسـودَ لا يُـسـامُ تروم المكر بالاسـلام مهـلا ... فحول القوم تعلـم مـا يُـرام تسمي عجزكم عن حرب قومي ... سلاما ..خبت مـاذاك السـلام فليس السلم يا مغـرور قـولا ... تخالفـه المجـازر والطـوام أيا أوبام لا يغـررك ضعـف ... أصـاب المسلميـن فـلا دوام تستَّرتـم بأسـمـالٍ خِــلاق ... فعرَّاكم مـن الصـدق العـوامُ خفضتَ الرأس مـن ذلٍ إلينـا ... و كان خضوعكم هونا يُسـامُ رأيتَ الحرب أعيتكـم رحاهـا ... وكلّت عـن ادارتهـا الطغـام فجئت مضعضعا ترجوا فكاكـا ... وليس لديك في نصـر مـرام تحزَّبتم بأحزابٍ كـلابٍ علـى ... الإسلام مـن حنـقٍ تحامـوا فصدّتها أسود عـن شراهـا ... وحرَّق جيشكـم منَّـا ضـرامُ خسأتم ليـس ينفعكـم سـلامٌ ... فذوقوا مرَّ ما يلقـي الصِّـرامُ نتاج الحرب يـا هـذا قبيـحُ ... وإن غشَّـى قباحتـه الظـلام أتوا بك أسودا من أصل خيـرٍ ... لتجرع كأس ذلِّتهـم وصامـوا أتوا بك خرقة عَرَكـت خبيثـا ... من الآثار واستلقـوا ونامـوا ستعلم ما أُريـدَ أيـا أوبامـا ... إذا سحب الجيوش غدا سـلام ستعلم أيهـا المسكيـن حتمـا ... بـأن مـرادهـم ســمٌّ زؤام وأنك لعبـة حيكـت لمغـزى ... ودُبِّر ما جهلـت أيـا غـلام إذا نجحـت مكيدتهـم تنـادوا ... بأن الحكم فـي يدكـم زمـام وإن فشلت خديعتهـم تداعـوا ... إلى خلعٍ بـه الإسـلامَ لامـوا دهتكـم أزمـة ونمـا وبـاء ... وتحريق الجيوش لكـم حِمـام تصيبكم المصائب كـل حيـن ... عـذاب الله.. صبَّـابٌ غـرام ستهزمكم مـن الأجنـاد جنـد ... على الإسلام قد حرصوا فقاموا ستُرفع راية الإسـلام حتمـا ... وتحقيق السـلام لنـا وسـام
