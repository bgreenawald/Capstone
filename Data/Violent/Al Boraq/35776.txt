قبِّل كــــفَّ بابــَـاهـُــم.... هم الأعــداءُ من صِغر عرفنــاهم و لكـِـنَّا نسينــا أنــنا صرنا أُسارى من أساراهم وأنـــا بعد أيـــامٍ سنصبح من رعاياهم و نهديــهم سيوفاً ضرجت بالامس من أشلاء قتلاهم و نهــديهم تــراب القـــدس و المسرى الذي قد صار مســراهــم و نذهب عندهم جـــريا لنجلــس عند بابـــاهم أميـــــر القــــوم.... هل للقــــوم سهـــم من هدايــاهم نبايـــعكم....!! و بيعتــنا عقــدنــــاها برومـــاهم فيا عاري..... و يا ذلــي..... بمــاذا سوف نلقــــــاهم أميري سيدي شيخي لماذا سرت نحو الغرب كي تمحوا خطاياهم تراه تخاطب الاديــان مزقـــنا و أبقــــاهم أتنـــزلنا لهــــذا القــــدر... لتـــركبنا مطايـــاهم... أتركعنـــا لترضيـــهــم و تهجــــرنا و تغشاهـــم و أرضي هدهـــا ظمـــأ و نهر سِقــاك روَّاهـــم مـــاذا قلت.......؟ مــــاذا قال......؟ مـــاذا.........؟ أنت.. مــــاذا...؟ هـُــم... نعـم يــا من تبــادلتــم... كلاما عنك أرضاهم فـَـداءُ القــومِ نعــرفه جــــهـــاداً قبـــل داواهـــم خلــوت... و خابت الخـُـلــوات و مـــن بالســـر ناجـــاهم ستجلــس فــوق تاريخ لكـــم خطــته كفـــاهم و تبقــى حاسراً تشكوا إذا انسحبــت ســراياهــم سجيـــن زيـــارة البـــابا قعــيــــداً فــي زوايـــاهم و تبصر و جهــنا المجروح ينـــزف في مرايـــاهم أنا لا زلــت لم أنطـــق ببـــلواهــــم أنـــا ما زلــت لا أدري.. لمـــاذا الكـــل يهـــواهم و أنت و ذلك البابــا و حكــام عرفنـــاهم أحدثكم حديث النفــــس يا مــن بـــان مسعـــاهــم حسبت سيــوفــكم شــحــذت لتلقـــاهــــم لتنصــــــرني و ابنــائـــي و أحبــاب تـــركنـــاهم فكـــان الســيف في حلقـــي و قـطـعنـــي و أرضــاهم فلا تحزن كبيــر القـــوم .... لا تغضب إذا أبكـــيت ليــلانـــا .. فقـــد أرضيت ليــلاهــم عجبت من الذي يجري لمـــاذا الـــكل يــخشـــاهم لماذا بِـتُّ أكرهــهــم........و أكــره مــن تــولاهــم و أكــره كل ما يأتــي.... إلينــا من عطــاياهـــم سئــمــت الدمــع لو يجــري لمجــراهم سئمت الأمــس و الذكرى التــي تأتــي بذكراهم كبــيـــر القـــوم... دع رومـــا و لا تــقـبــل تحاياهـــم و لا تجـــلــس على قـبــري تقــبــل كـــف بــابــاهـــم.. .. .. الجبـــــوري
