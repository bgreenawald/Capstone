حـكامــنا دخـل الزعيـم المنتخب..بطريقـة ديمقراطيـة لـيـس عليـهـا غـبـار.. دخــل مــع مـواطـن مــن عـامـة الشـعـب فــي حـــوار .. و رفــــض الـمـواطــن ذلــــك و قـــــدم الأعـــــذار. قــال لا استطـيـع أن أحــاورك أنــت الزعيـم..وأنـا حـــمـار .. فــغــصــبــه..بـــإرادتـــه الـــكـــامــــلــــة.. قـــال أريـــد أن اعـــرف شـكــاوى الــيــد الـعـامـلـة.. فأذعن المواطـن قائـلا.. بشـرط أن لا انتهـي لمصيـر جـاري عمـار.. وكان الجار قد قضى في عراك.... بالرصاص.. مشنوقا ...و سجل انتحـار.. قــــــال الـزعـيــم..لــك مـــنــــي الأمـــــــان.. فــتــكـــلـــم يـــــــــــا حـيـــــــــــوان.. بـدأ المواطـن قـائـلا.. فــي بـلادنـا الخـبـز مرتـفـع الأسـعـار.. رد الـزعـيـم قـائــلا.. &quot;لا للسـمـنـة&quot; ذاك هــــو الـشـعــار.. - البـطـالـة متفـشـيـة.. و أثـقـلــت الـشـبــاب الـفـاقــة.. - عجبـت مـن قومـي.. أرحتـهـم.. فطالـبـوا بالأعـمـال الشـاقـة.. - والكلام ممنوع.. والنقد مرفوض.. و بيني و بينك ألـف جـدار و جـدار.. - دعـك مـن قيـل و قـال.. فللسياسـة يـا غشيم..متاهـات و أسـرار.. - الـسـجـون مكـتـظـة.. بـأبـنــاء الـشـعــب الــبــريء.. - شكـوتـم أزمــة السـكـان.. فـكـان هــذا الـحـل الـجــريء.. تعجب المواطن من ثقـة الزعيـم بنفسه..قـال انـك يـا سيـدي مظلـوم.. و مــن تكـلـم عـنـك بالـشـر يسـتـحـق الـمــوت مـرجــوم.. ولكن.. بقي سؤال أخيـر.. و استسمـح سعادتـك عـذرا علـى التأخيـر.. قال سـل و لا تخـف.. أنـت مـن الرعيـة.. ولـك منـي كـل التقديـر.. قال.. في بلادنا.. أنا مثلا.. لا يمكـن لـي أن أكون..مثلـك زعيـم الديـار.. فأمـر بتعليقـه علـى عـود المشنقـة.. ووجـد نفسـه مكـان عـمـار.. وودعــــــه الــزعــيــم بـــصـــوت شـــجــــي.. أنـــت الآن ذاهـــب لـربــك.. فـسـلـه يــــا غــــبـي.. لم لا تستطيع أن تكون إلها.. مثله..جبار..
