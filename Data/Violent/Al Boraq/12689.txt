بسم الله الرحمن الرحيم يا مستورين ستكشف الخطايا لا تغفل عن المــــــــــــــــــــــــــــــــــــــــــــوت لاتغفل عن المـــــــــــــــــــــــــــــــــــــــــــوت ليس له مكان معــــــــــــين ولا زمن معـــــــــــــين ولا سبب معين ولا عمر معين يأتيـــــــــــكم :::: بغتـــــــــــــــــــــــــــــــة! وأنتم لا تشــــــــــــعرون (( وجاءت سكرة الموت بالحق ذلك ماكنت منه تحيد)) إستعدوا للموت قبل أن ينادى بأعلى الصوت رباه إرجــــــــــــــــــــــــــــــعون فلا يســــــــــــــــتجاب له (( حتى إذا جاء أحدهم الموت قال رب إرجعون)) اللهم إنك تعلم أنى قد أخافك أنا الآن أرجوك اللهم إنك تعلم أنى لم أكن أحب الدنيا لطول البقاء ولكن لصيام الهواجر وقيام الليل ومجالسة العلماء _- إنظر على ماذا يبــــــــــــــــــــــــــــــــــــــــــكون ! إنظر على ماذا يتحـســــــــــــــــــــــــــــــــرون تعال وإنظر ياغافل تعال وإنظر من هم التوابون غيرك يتوب ويرجع إلى ربه وأنت مازلت تتكبر ..: وعلى من ؟؟ أجب أيها العاصى على من تتكبر أنت ؟ على ربك ليتك تصدق بأنك ستموت اشغلتك الدنيا بمالها ومتاعها وما أوجدها لك غير خالقك وأخيرا ( سبحانك اللهم وبحمدك أشهد أن لا إله إلا أنت أستغفرك وأتوب إليك)
