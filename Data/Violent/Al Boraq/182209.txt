أبيات للدفاع عن أمنا عائشة رضي الله عنها وأرضاها لبـــــــيك يا زوج الحبيب محــمدا***أماه كل المسلمــــــــين فـداكِ لبـيك أبناء العقيدة كلــهــــــــــــم ***عزموا على نـحر الــذي آذاكِ يا زوجة الهــادي إلى سبل الهدى*** بيت النبوة مرتعا لتقـــــــــاكِ مات النبي وكان عنــــك راضيـا*** خير النساء بــــذاك قد زكّاكِ خاب الكفور الخاسر المتشــــــدّق***فالله رب العرش من بـــــرّأكِ كفر الخبيث مظاهرا بلسانـــــــــه***خسر اللعين الكاذب الأفّــــاكِ الله أدعو أن يعــــــــجل ذبـــــحه***وفصاله مما يلي الأفّكــــــاكِ هجر الذليل لكي يبـوح بكفـــــــره***بين الصليب وجملة الضّحّاكِ حمل العداء لأعظم الناس هــــدى***نهق الخسيس بفاهه الهّـــتّاكِ هو والذين يــــــــؤيدون فعالـــــه***النار مثواهم مع الهــــــــــلّاكِ يصلى سعيرا في الجحيم مقــــيدا***بسلاسلٍ يُحمى على الأشّـواكِ تالله لــــــــــــــو لاقيتــــه لقتلتـــه***ونفذت فيه الخنجر الفتّــــــاكِ أبشر بسلخك أيــــــها الكلب الذي***يعوي بمخبأ ملتقى الأدّيــــاكِ يا أيها النذل المعمم إننـــــــــــــي***أتٍ إليك وخلفي الأفّــــــــلاكِ آتٍ أزلزل فأستعد لضربتـــــــي***فلسوف تندم راجيا لفـــــــكاكِ هيهات لن أرحم مجوسا خائـــــنا***بل سوف أقطف رأسه السّفّاكِ الرافضي نتنٌ حقيرٌ شأنــــــــــــه***مهما رأيت عمامة وســـــواكِ عبد الحجارة خاضعا لها ساجدا***عجبا ويلطم في المحافل باكِ أما الزنى أفتى به فأباحــــــــــــه***ويرى بذلك انه متــــــــــــذاكِ يا ويله هو كافـــرٌ بل جاحــــــــدٌ***بل مشــركٌ متفرع الأسّـــلاكِ يختار من دين اليـــــهود عيـيـنـةً***وكذا من الصلبان والشّــكّــاكِ يارب أنزل بالروافض حتفهـــــم***واحرمهم الألقاب والأمــــلاكِ تم الإنتهاء من من تكملة الأبيات يوم الأربعاء الموافق 6/10/2010 ندعوا الله أن يغفر لنا ويحشرنا في زمرة الذين ساهموا بالدفاع عن أمنا عائشة رضي الله عنها وأرضاها وعن وصحبه الكرام رضوان الله عليهم أجمعين .. أتمنى نقدا بناء عن القصيدة إن تطلب ذلك وأكون شاكرا لكم
