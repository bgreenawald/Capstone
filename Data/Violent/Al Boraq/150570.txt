العراق المقاوم1-الصدق ربّانا بمقودِ مركبـــــــــــــــــي فبه المخاطر و المكاره تنجلـــــي 2-إن زل نفسا أو أراد عداوتــــــــــــــي يبقى السبيل لردع جلّ مظالمــــــي 3-الخير يأتي إن سعيت لكسبـــــــــــــــه والحق من يسلك يفوز ويهتــــــدي 4-أنعي كرام القوم حين ترحّلـــــــــــــوا وفقيد أمتنا الذي لا ينحنــــــــــــــي 5-هم أوثقوه بالحبال وأحكمـــــــــــــــوا بزعيمنا المقدام صدام الأبـــــــــي 6-رفض القناع فخاف كل مقنـــــــــــــعٍ والموتُ منه كاد يدبر منثنـــــــــــي 7- صدام ليثا للشجاعة رائـــــــــــــــــدا شهما كغيثٍ بالكرامة يرتقــــــــــي 8-كان الرخاءُ مع العطاء بعهـــــــــــده والأمن ينعم من أتاه ويرتــــــــــوي 9-رمز العراق وتاج أرض المشـــــرق ما مات بل في كل قلب يرتعـــــــــي 10-أخذ البطولة ماسكا بزمامـــــــــــها حاز المكارم بالفضائل يعتلــــــــــي 11-أرض العراق طغاة عصرك ألفظي وامضي على درب الرسول المقتدي 12-ودعي الذين تهاونوا عن نصـــرك أو من يُعوّق في الزحام ويختفـــي 13-أو من تمسك بالحياة مصمـــــــــمٍ ألف التخاذل خلف جدر ينطــــــوي 14-لا خير فيمن بالأخوة يدعــــــــــــي ويعين طاغوتا يدنس تربتـــــــــــي 15-عجبا لمن باع البلاد بدرهــــــــــمٍ ترك الهوية خالعاً ما ينتمــــــــــي 16-النصر ينكر من يخون الموطـــــن أو مرجفاً و بكل كفرٍ محتمـــــــــي 17-جند العراق أيا أُسوداً زمجـــــري دكي الجبال بملْ صوتك زلزلــــــي 18-هزي بريحٍ صرصرٍ متوحــــــــــدٍ حصناً يسيّسه العميل المرتشـــــي 19-دوسي العدو بكل ثقلٍ أسحقــــــي واسقي المنية للظلوم المعتـــــدي 20- يا جيش إسلامي فداكم مشرقي فالغرب أصبح كل يومٍ يكتـــــــوي 21-أنصار سنتنا وحرمة مسجــــــدي صفوا الذين يدنسون مصاحفــــــي 22 وخذوا بثأر بنيةٍ من طائــــــــــفي من ساقطٍٍ لبّ النذالة يرتــــــــــدي 23-لكم التحية يا حماة المسكــــــــــن وحماة أطفال العراق المفتـــــــدي 24-أنتم أمل شعبا وعزة موطنـــــــي فبنهجكم تسموا الشعوب وتقتــدي 25- الدمع يجري لا يفارق مقلـــــــتي والشوق في وصل الجهاد ألم بـــــي 26-لولا الحدود وما يعارض وجهتـــي لأتيت أخدمكم وكنتم قبلتـــــــــــــــي 27-مزن السماء إلى العراق تجهزي سيري بإذن الله لا تتوقفـــــــــــــي 28- صبحا أقيمي في رباه وامطـــري ماءً يروي الأرض حتى تكتفــــــي 29-زوري قبور الصالحين وقبّلــــــي عنا شهيدا طاهرا فتشهـــــــــــــدي 30- هذا وصلوا للحبيب محمــــــــدٍ والحمد لله العظيم المعتلــــــــــــــي
