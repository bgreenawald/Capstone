بـسـمـ الله الـرحـمـنـ الـرحـيـمـ عـن أبـي هـريـرة رضي الله عـنـه قـال : قـلـت يـارسـول الله مـن أسـعـد الـنـاس بـشـفـاعـتـك يـوم الـقـيـامـة ؟ قـال رسـول الله صلى الله عـلـيـه وسـلـم { لـقـد ظـنـنـت أن لا يـسـألـنـي عـن هـذا الـحـديـث أحـد أول مـنـك لـمـا رأيـت مـن حـرصـك عـلـى الـحـديـث } { أسـعـد الـنـاس بـشـفـاعـتـي يـوم الـقـيـامـة مـن قـال : لا إلـه إلا الله خـالـصـا مـن قـلـبـه أو نـفـسـه } أخـرجـه الـبـخـاري / 99 وعـنـد الـطـبـرانـي فـي الأوسـط عـن زيـد بـن أرقـم مـرفـوعـا ( مـن قـال لا إلـه إلا الله مـخـلـصـا دخـل الـجـنـة ) قـيـل : مـا إخـلاصـهـا ؟ قـال : ( أن تـحـجـزه عـن مـحـارم الله )
