( عُمــر سلَيمان ) مُديرِ المخـَــابَراتِ المصْريّة ليُوافقَ ويوقِّع بِنفْسـِــه علَى اتّفاقِ التّهــْـدِئة الأوّل في شهـْــرِ حُزيران 2008م ، قائِلاً بالنّـــص حينَما سُأِلَ عن رأْيـِــه فِي التهدِئَة قَال : ( فِش مُشـــكلة .. حبيبي يا حـــج ) ، ولعلّ فــِـي هَذا إزالةً لِلغُمـــوضُ الّذي لفّ موقِفَ ما يُسمّـــى بِـ ( جيْش ِالإســـلاَم ) طِيلةَ التهدئَة السّابقَة ، فكلّــــهُم وقّعوا عليْها صَاغـِــرِين ، وإسْألـُــوا مُمتَاز ، بَل والجيـْــشَ كلّه ، أيـْــنَ كانَ حين قُتِل السّبــعةِ الّذين يدّعــُـون أنّهم مِنَ الجيْش ، وأترُكِ الإجـَــابَة لهُم إنْ كـَــانَ لهُم إِجابــَـة ، أمــَّـا عَن شلّة خانْيونـُــس الجَديدة الّذينَ يسَمـّــون أنفُسهَم ( جَماعةِ التّوحِيـــد وَالجهَاد ) فقد بَدأوا مِشـْــوارَهم ( الجِهَادي ) بالْكـَــذِب ، وهُو سنّة جارِيـَــة بيْنهُم مِن بَاب ( الحَربُ الإعلاَميـّــة! ) ، فَقد بثّت لهـُــم جَبْهة الخـِــدَاع العَالميّة شَرِيطاً لِما قَالـُــوا أنَّها عمَليّة مرئيّـــةٌ لهُم ، هكَذا وبِكلّ بسـَــاطة صَدّق خِرفانُ الإنْترنـِــت هَذا الشّريط ، وأقْسـِــمُ أنّ أحدَ الإخوَة رآه فِي اليوم التّالــِـي للعملِيــّـة على قنَاة إسـْــرائيل الثانِيَة ، وأنا شـَــاهدتُه قبل نُزولِ العمَليــّـة بأسْبــُـوع تقريباً ، وشـَــاهدتُ المادَة الخَام للعمليّة على اليُوتيـــوب ، فكَفـــاكُم كذباً ، أكْثـَــرُ ما يُثيرُ الضّحك فِي المــَـوضُوع أنّ أحدَهُم أثـَـــار هذِه القَضيّة ، فكــَـان التّبرِيـــرُ جاهِزاً مِن أحَد الخـِــرفَان بأنّ ( الشّرِيــــطَ ضاعَ قَدراً مِن الأخِ المُصوّر ووَصـَــل إلَى مُراسِل إسـْــرَائيلِ الثّانية وبثّتــْـهُ القنَـــاة !! ) ضـَــاع ، ووصَل فِي أقلِّ مِن 24 سـَــاعةٍ لهُم ، ثمّ حصلْتـُــم أنْتُم علَيْهِ مرّة ثَانيةً بِقـُــدرة العزِيز العَليِم !! ، للـَّــه دَرّ الكَذِب مَا أعـــدَله بَدأ بِصـَــاحبِه فكَشفـَــهُ ، ولوْلاَ كَــــرَامةُ المُنْتدى لاسْتخْدمتُ عِبـــَـاراتٍ مِن عِِيار ( وَجيه عبّاس ) و ( أُســَــامَة فوْزِي ) مع الخِرفــَــان الجُدد ، تُذكّرنــِي هَذه الشلّة الجدِيـــدَة بـ ( شِلــّة ) سَبقتـْــهَا في الكَذِب اسمُهَــا ( جُند اللّـــهِ في فلَسـْـطِين ) ، أتمنَّى من سِينَمــا ( خَانْيونـُـس ) أنْ تخفّض سِعرَ التّذاكِر المجـِّـانيّة لمتَابعةِ مْسرحيـَّـاتِها ! ، قبْلَ أن أنهـِــيَ الجُزء الأوّل من هَذا المُسلسـَــل ، والّذي أكْتُبه قبـْــلَ أن أكمِلَ حلقـَــات (اللّيلـَـــة الّتِي صارَ فيها صدِيقي &quot; عـَـــاقِلاً &quot;! ) الّتي تأخّـــرت كثِيراً ، وكان يجِـــب أن تنْتهِي لوْلا مـــشاغِلي ، أود أن أذكر عدّة أشْيــَــاء : أوّلاً : تَهنِئـَـــة (خاصّة) من طَرفِي بمُناسَبـــةِ الإفْرَاج عنِ (شاعِـــر القَاعِدة) المُهنْدِس الأردنـِـي ( مُحمّد الزُّهيـْــرِي ) ، وعوْدتِه بمَجمُوعة معرّفــَـاته (كَتيبةُ الأهْوالِ ، الكَتيبةُ الخَرساء ، مَنْ أنت) . ثَـــانِياً : تَهنئةٌ (خـــاصّة) أيْضاً للعالِم الّـــذِي فُوجئْتُ بعلْمِه للعُضوِ (rafed) في الفلـــوجة أو ( شــاعِرُ الأنصَار ) فِي الإخـْـلاص بمُناسَبة حُصـــولِه علَى لقبِ ( الشّيـْــخ ) ، وأقولُ لَـــه : دعكَ من سَـــالِفة ( أبُو دُجــَـانّة الأزْدِي - الشـّـرق الأعلَى ) ، فالسّـــالفة لمْ يعُد فِيها لا شـَــرق أعلَى ولا شرقٌ غـَــربيّ! . ثَالثــاً : حمّلنـــِي أِخي أبُو إسلام وَهُو نِيَابةً عَن ( جِهـَـــاز الأَمنِ الدّاخِلي ) رِســـَالة شكْرٍ لكُلٍ مِنَ الأعضـَــاءِ في مُنتديَات الفَلوجـَــة ( أبُو جعـــفَرِ الشّامي ، المَاجِستيـــر ، فِتْية الكَهـــفِ ، مُحمّد المسلـــم ، أبُو محمـّـــد المقْدِسي ) علَى المعلُومات القيّـــمة الّتِي ذكرُوها عنِ المُعتَقــــلِ لدَى المبـَــاحِث : ( أحمَد شبــَــات ) ، فَما كَان الأمـْـــنُ الدّاخلِي ليحَصلَ علَى هذِه المعلُومَات لوْلا تَعــــاوُنِكم الّذِي يَتمّنـــوْنَ مِنكُم أنْ تَستمرّوا بــــه . ويستمرّ مُسلســَــل الفَضائِح ،،
