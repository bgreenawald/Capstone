) ضـَــاع ، ووصَل فِي أقلِّ مِن 24 سـَــاعةٍ لهُم ، ثمّ حصلْتـُــم أنْتُم علَيْهِ مرّة ثَانيةً بِقـُــدرة العزِيز العَليِم !! ، للـَّــه دَرّ الكَذِب مَا أعـــدَله بَدأ بِصـَــاحبِه فكَشفـَــهُ ، ولوْلاَ كَــــرَامةُ المُنْتدى لاسْتخْدمتُ عِبـــَـاراتٍ مِن عِِيار ( وَجيه عبّاس ) و ( أُســَــامَة فوْزِي ) مع الخِرفــَــان الجُدد ، تُذكّرنــِي هَذه الشلّة الجدِيـــدَة بـ ( شِلــّة ) سَبقتـْــهَا في الكَذِب اسمُهَــا ( جُند اللّـــهِ في فلَسـْـطِين ) ، أتمنَّى من سِينَمــا ( خَانْيونـُـس ) أنْ تخفّض سِعرَ التّذاكِر المجـِّـانيّة لمتَابعةِ مْسرحيـَّـاتِها ! ، قبْلَ أن أنهـِــيَ الجُزء الأوّل من هَذا المُسلسـَــل ، والّذي أكْتُبه قبـْــلَ أن أكمِلَ حلقـَــات (اللّيلـَـــة الّتِي صارَ فيها صدِيقي &quot; عـَـــاقِلاً &quot;! ) الّتي تأخّـــرت كثِيراً ، وكان يجِـــب أن تنْتهِي لوْلا مـــشاغِلي ، أود أن أذكر عدّة أشْيــَــاء : أوّلاً : تَهنِئـَـــة (خاصّة) من طَرفِي بمُناسَبـــةِ الإفْرَاج عنِ (شاعِـــر القَاعِدة) المُهنْدِس الأردنـِـي ( مُحمّد الزُّهيـْــرِي ) ، وعوْدتِه بمَجمُوعة معرّفــَـاته (كَتيبةُ الأهْوالِ ، الكَتيبةُ الخَرساء ، مَنْ أنت) . ثَـــانِياً : تَهنئةٌ (خـــاصّة) أيْضاً للعالِم الّـــذِي فُوجئْتُ بعلْمِه للعُضوِ (rafed) في الفلـــوجة أو ( شــاعِرُ الأنصَار ) فِي الإخـْـلاص بمُناسَبة حُصـــولِه علَى لقبِ ( الشّيـْــخ ) ، وأقولُ لَـــه : دعكَ من سَـــالِفة ( أبُو دُجــَـانّة الأزْدِي - الشـّـرق الأعلَى ) ، فالسّـــالفة لمْ يعُد فِيها لا شـَــرق أعلَى ولا شرقٌ غـَــربيّ! . ثَالثــاً : حمّلنـــِي أِخي أبُو إسلام وَهُو نِيَابةً عَن ( جِهـَـــاز الأَمنِ الدّاخِلي ) رِســـَالة شكْرٍ لكُلٍ مِنَ الأعضـَــاءِ في مُنتديَات الفَلوجـَــة ( أبُو جعـــفَرِ الشّامي ، المَاجِستيـــر ، فِتْية الكَهـــفِ ، مُحمّد المسلـــم ، أبُو محمـّـــد المقْدِسي ) علَى المعلُومات القيّـــمة الّتِي ذكرُوها عنِ المُعتَقــــلِ لدَى المبـَــاحِث : ( أحمَد شبــَــات ) ، فَما كَان الأمـْـــنُ الدّاخلِي ليحَصلَ علَى هذِه المعلُومَات لوْلا تَعــــاوُنِكم الّذِي يَتمّنـــوْنَ مِنكُم أنْ تَستمرّوا بــــه . ويستمرّ مُسلســَــل الفَضائِح ،،حمداً لله على سلامتك أخي المـهنّــد، مشكور على المعلومات القيّمة. اقتباس: كاتب هذه المشاركة : المهنـــد ، ، وضَربـــاتٌ ( تَحت الإعدَاد ) فِي أمْرِيـــكا ، وأخْرَى في أورُبــّـا ، و ثَالثةٌ في مِصـْــر ، ورَابعةٌ في لِيبيــَـا ، وخَامسةٌ في اليَمـَــن ، وسَادسةٌ في أفْريقيــَـا ، وسابـِــعةٌ في الإِمَارات غريب، نسخة طبق الأصل عن لائحة الأهداف الصفويّة الجاري العمل عليها و على رأسها مصر(!!!!!)، طبعاً ما عدا أمريكا و أوروبّا الحليفتين.

يا أخــــِـي عُموم مسيرة الجهاد في بــِـــلاد الرافدين أستطيع أنْ أقول لكَ عنـــــها (الله يرحَمـُـــها) ، و (كانتِ المرحـــُــومة بنْت حلال) ! فبعْدمـــــا أحسّ ذاك الأمـــِـير المجهُول (أبو عمر البغـــْــدادي) بـ(الخــــازُوقِ) الذي شَرِبـــــه أتانا بـ(حصَاد الخيـــْـــر) ، فأيّ خيرٍ يا هذا والسمّ مــــَـا زال ينفث من بين أسنانِك بأنّ كــــــلّ من تخلّف عن بيعتِك هـــُــو (عاصٍ) ، وصِبيـــــانك يضعُون العبوات اللاّصقـــــة ضد عناصِر الحزب الإسـْــــلامي في بغداد والمُوصــــــل ، ولم تَفْتـــــأ تُهاجم كلّ من اختلف معــــَـك ، حتى وصلتْ تهجّماتك على حـــَــركة حَماس في فلسطين ، ! ] البـــَـــاحِث عن الحقيقة ، على فـِــــكرة ، الكلام اللّــــــي فُوق موجه لك (: بالنّسبــــــةِ لـ(مجموعةِ الأمّــــــة) ، فكلمة (جيــْــــش) تُوحي لكَ أن تعداد عناصرهم يفوقُ العشــْـــرة آلاف وهم لا يملؤون بــَــــاص 60 راكب ، فإِن كانــــُــوا قد شاركوا في الجهاد في الحـــــَــرب ، فبارك الله فِي جهـــــادِهم ، وهم أسيـــــادُك لوحدِك ، ولا تفْرِضهــــــم أسياداً على الآخرِين ، فمنِ ارتَضى أن يــَـــــكون عبْداً فهذا شأنُــــــه ! الغَريــــــب أن مصدَرك في الكــــــلامِ هو مراسل الجزيــــــرة الّتي تتهجم عليـْـــــها في السّطر التالي لكلامِـــــك ، فـ إِرْسَى على بـــَــــر هي جزيرة ولا خنْزِيـــــــرة (: ؟ بُخصوص حــــــَركة الشباب ، ألم يُعلنوا في بيان لَهم ذات مرّة أن مطــــــَار مقديشو مغلق بشكلٍ تــــــام ، مهدّدين كلّ الطــــــائِرات التي تُحاوِل الهبوط أو الإقـــــْلاع منه بالقصف والاسقــــــاطْ ، فكيْف استطـــَـــاع شيخ شَريف أن ينزل في المطــــــار ، وتعقِد حكومته اجتماعاتـــِــها ومؤتمراتِهـــــا الصّحفية في مَقديشو التي تدّعـــــــي حركة الشباب أنها حررّتــــــها !؟ عُمومـــــاً النّقاش في الوضع الصومالـــِـــي ليس مكانُه هنا ، وإن كنتُ أبـَـــــارك للمجاهدين الصادقيــــــن هناك دحر الصليبيين الأثيوبييّــــــن ، على ذِكــــــر الصومال ، فقد وضع أحدُ الخرفـــَـــان خبراً عن أنّ ( الحكيـــــم الظواهري ) شـُـــــوهِد جنوب الصومال (: (أيكُون المؤمِن كذاباً !!!) إســـــْـأل أبا عُمر البغدادي الّذي زلزل عـــــَـالم الإنترنّت بخطابٍ يتوعّد فيه إِيــــــران (: وراجِع كمّ الأكاذِيـــــــب التي أدْرجها أسد البطيخ2 في أجوِبتـــــــه على أسئلتكم ، ومن قبــــــلِه يمان مخضّب ، وغيــــــرهم ، ولستُ في معـــْــرض الدفاع عن فارس الجنـــُـــوب ، ولكِن اتّعظــــــوا من الفضيحة بِجــــــلاجِل اللي سوّاها موضُوعــــــه فيكم ، عسَى أن تتطور عقولــُــــكم قليلاً ! نَسيت أن أذكـــُـــر شيء : إلا شــــُــو قصة الحزبيّة اللي ماسِكْلــــــي فيها بإيديك وسنَانك !؟ أم أنّـــــها التهمة الجاهِزة لكلّ من يُخــــــالفكم الرأي ! إِذا كنتَ تُلمّح لموضـــُـــوع أني أنْتمي لحركة حمــــَــاس ، فراجِع موضوعيَ جيّداً الذي قمتَ أنتَ برفْعـــــــه ، مشكُوراً ، بخصوص مُحــــــاولة اغْتِيال بلير ، وستجِد الرّد على اتّهـــــامكَ السخيف هذا هُناك (:
