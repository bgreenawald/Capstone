بسم الله الرحمن الرحيم الحمد لله ناصر عباده الموحدين و مذل اهل الشرك والكفر اجمعين والصلاة والسلام على نبينا الامين محمد خاتم النبيين وعلى اله و صحبه اجمعين و بعد : فقد قال تعالى: ( يَا أَيُّهَا الَّذِينَ آمَنُواْ اتَّقُواْ اللّهَ حَقَّ تُقَاتِهِ وَلاَ تَمُوتُنَّ إِلاَّ وَأَنتُم مُّسْلِمُونَ) [آل عمران : 102] وقال تعالى: ( يَا أَيُّهَا الَّذِينَ آمَنُوا اتَّقُوا اللَّهَ وَقُولُوا قَوْلاً سَدِيداً) [الأحزاب : 70] وقال تعالى : (يَا أَيُّهَا النَّاسُ اتَّقُواْ رَبَّكُمُ الَّذِي خَلَقَكُم مِّن نَّفْسٍ وَاحِدَةٍ وَخَلَقَ مِنْهَا زَوْجَهَا وَبَثَّ مِنْهُمَا رِجَالاً كَثِيراً وَنِسَاء وَاتَّقُواْ اللّهَ الَّذِي تَسَاءلُونَ بِهِ وَالأَرْحَامَ إِنَّ اللّهَ كَانَ عَلَيْكُمْ رَقِيباً) [النساء : 1] فاعلموا هداكم الله ان دين الاسلام يوجب الاستسلام لله بالتوحيد والانقياد له بالطاعة والبراءة من الشرك واهله. وان الله خلق الانسان لعبادته وهذا هو سر الوجود قال تعالى : (وَمَا خَلَقْتُ الْجِنَّ وَالْإِنسَ إِلَّا لِيَعْبُدُونِ) [الذاريات : 56] وقد ارسل الله الرسل والانبياء للدعوة الى عباده الله وحده وعدم الاشراك به شيئا وقد بين الله ذلك في مواضع كثيرة في القران فقد قال تعالى : (لَقَدْ كَفَرَ الَّذِينَ قَالُواْ إِنَّ اللّهَ هُوَ الْمَسِيحُ ابْنُ مَرْيَمَ وَقَالَ الْمَسِيحُ يَا بَنِي إِسْرَائِيلَ اعْبُدُواْ اللّهَ رَبِّي وَرَبَّكُمْ إِنَّهُ مَن يُشْرِكْ بِاللّهِ فَقَدْ حَرَّمَ اللّهُ عَلَيهِ الْجَنَّةَ وَمَأْوَاهُ النَّارُ وَمَا لِلظَّالِمِينَ مِنْ أَنصَارٍ) [المائدة : 72] و قال تعالى : (لَقَدْ أَرْسَلْنَا نُوحاً إِلَى قَوْمِهِ فَقَالَ يَا قَوْمِ اعْبُدُواْ اللَّهَ مَا لَكُم مِّنْ إِلَـهٍ غَيْرُهُ إِنِّيَ أَخَافُ عَلَيْكُمْ عَذَابَ يَوْمٍ عَظِيمٍ) [الأعراف : 59] وقال تعالى : (وَإِلَى عَادٍ أَخَاهُمْ هُوداً قَالَ يَا قَوْمِ اعْبُدُواْ اللّهَ مَا لَكُم مِّنْ إِلَـهٍ غَيْرُهُ أَفَلاَ تَتَّقُونَ) [الأعراف : 65] و قال تعالى : (وَإِلَى ثَمُودَ أَخَاهُمْ صَالِحاً قَالَ يَا قَوْمِ اعْبُدُواْ اللّهَ مَا لَكُم مِّنْ إِلَـهٍ غَيْرُهُ قَدْ جَاءتْكُم بَيِّنَةٌ مِّن رَّبِّكُمْ هَـذِهِ نَاقَةُ اللّهِ لَكُمْ آيَةً فَذَرُوهَا تَأْكُلْ فِي أَرْضِ اللّهِ وَلاَ تَمَسُّوهَا بِسُوَءٍ فَيَأْخُذَكُمْ عَذَابٌ أَلِيمٌ) [الأعراف : 73] وغير ذلك من ايات كثيرة. فأعظم ما امر الله به العبد هو توحيده سبحانه واعظم ما نهى الله عنه هو الشرك بالله. و بعد هذه فاعلموا هداكم الله ان النبي محمد ( صلى الله عليه وسلم) قد دعا الكافرين الى الاسلام و جاهدهم باليد واللسان والقلب. وقد عانى النبي ما عانى من جوع وعطش واذى والم وحزن قاساه هو واصحابه و كل ذلك في سبيل الله ولم يخشى النبي او اصحابه في الله لومة لائم فضحوا بالغالي والنفيس في سبيل الله فهذا هو النبي صلى الله عليه وسلم قد لبس درعه وحمل سيفه واخذ جيشه ليجاهد الكفار من اجل اعلاء كلمة الله و ليكون الدين كله لله. وقد تمخض جهاد النبي صلى الله عليه وسلم ودعوته الى الاسلام بقيام دولة الاسلام التي اذلت اعتى الامبراطوريات في ذلك الزمان مثل امبراطوريات فارس والروم فانتشر نور الاسلام يجوب الارض وصارت الناس تدخل في دين الله افواجا. اناس صدقوا الله فصدقهم الله... قال تعالى : (مَن كَانَ يُرِيدُ حَرْثَ الْآخِرَةِ نَزِدْ لَهُ فِي حَرْثِهِ وَمَن كَانَ يُرِيدُ حَرْثَ الدُّنْيَا نُؤتِهِ مِنْهَا وَمَا لَهُ فِي الْآخِرَةِ مِن نَّصِيبٍ) [الشورى : 20] و قال تعالى : (وَلَقَدْ كَتَبْنَا فِي الزَّبُورِ مِن بَعْدِ الذِّكْرِ أَنَّ الْأَرْضَ يَرِثُهَا عِبَادِيَ الصَّالِحُونَ) [الأنبياء : 105] فهذا كتاب الله وهذه سنة نبينا عليه الصلاة والسلام من تمسك بهما نجى ومن خالفهما وتركهما وراء ظهره فقد ضل. و قد شرع الله الجهاد للاعلاء كلمة الله في الارض فيكون الحكم لله. فلا شرع الا شرع الله سبحانه فكيف يشهد مسلم انه لا اله الا الله و ان محمدا رسول الله ثم يحتكم الى غير كتاب الله و سنة نبيه و كيف يشهد المسلم انه لا اله الا الله و محمدا رسول الله ثم يوالي الكافر الذي يسب الله و يسب النبي. اذكروا هداكم الله قوله تعالى :( وَلَن تَرْضَى عَنكَ الْيَهُودُ وَلاَ النَّصَارَى حَتَّى تَتَّبِعَ مِلَّتَهُمْ قُلْ إِنَّ هُدَى اللّهِ هُوَ الْهُدَى وَلَئِنِ اتَّبَعْتَ أَهْوَاءهُم بَعْدَ الَّذِي جَاءكَ مِنَ الْعِلْمِ مَا لَكَ مِنَ اللّهِ مِن وَلِيٍّ وَلاَ نَصِيرٍ) [البقرة : 120] هذا في كتاب الله الذي لا ياتيه الباطل من بين يديه ولا من خلفه..بين الله لنا ان اليهود والنصارى لن يرضوا عن المسلمين فهم لا يكفون عن محاولة صدهم عن الاسلام و يكيدون ليل نهار للمسلمين لا يرضيهم الا ان ينقاد المسلمون اليهم و يرضوا بحكمهم و هذا والله لن يكون فما كان للمسلم ان يرضى بغير شرعه الله منهاجا و بغير القران والسنة دستورا ومنهجا للحياه. فيا كل من انتسب الى الصحوة في الانبار.. تذكروا قول الله سبحانه : (يَا أَيُّهَا الَّذِينَ آمَنُواْ لاَ تَتَّخِذُواْ الْيَهُودَ وَالنَّصَارَى أَوْلِيَاء بَعْضُهُمْ أَوْلِيَاء بَعْضٍ وَمَن يَتَوَلَّهُم مِّنكُمْ فَإِنَّهُ مِنْهُمْ إِنَّ اللّهَ لاَ يَهْدِي الْقَوْمَ الظَّالِمِينَ) [المائدة : 51] و هذا نص واضح و صريح على عدم جواز موالاة الكافر فلا يحق لمسلم ان يساعد الكافر في حربه ضد المسلمين فان فعل ذلك فهو كافر خارج من المله . و من مظاهر موالاتهم..مساعدتهم في خططهم التي اعدوها لغزو بلاد المسلمين فكل من ساهم في تحقيق غايات الكافرين في حرب الاسلام والمسلمين فهو كافر خارج عن الدين. و قد حدث هذا واضحا في الانبار التي كانت اعتى مدن بلاد الرافدين على الامريكان..فكان المجاهدون يقضون مضاجع الامريكان فيها و يدكون قواعدهم دكا واصواتهم تسمع من بعيد قائلة (الله اكبر ..الله اكبر) فأتى من بعد ذلك من زين له الشيطان حب الدنيا فبات يقول حان الوقت ليتوقف الجهاد لانه فتنة حسب زعمه فتناسى صاحب هذا الرأي قول الله تعالى : (وَقَاتِلُوهُمْ حَتَّى لاَ تَكُونَ فِتْنَةٌ وَيَكُونَ الدِّينُ لِلّهِ فَإِنِ انتَهَواْ فَلاَ عُدْوَانَ إِلاَّ عَلَى الظَّالِمِينَ) [البقرة : 193] فيبدوا واضحا من قوله تعالى انه لا يخلوا زمان من الفتن و ان الفتنة ليست سببا لترك الجهاد بل هي سبب للجهاد ... فيجاهد المسلم حتى لا يكون دين الا الاسلام. والرد على كذبه من ادعى ان الفتنة سبب لترك الجهاد بتذكيره بفعل صديق هذه الامة ابو بكر الصديق (رضي الله عنه) حين المت بالامة الاسلامية فتنة شديدة و كان مصير الاسلام برمته بخطر حين ارتد من ارتد عن الاسلام بعد وفاه النبي محمد (صلى الله عليه وسلم) و كانت هذه فتنة عظيمة الا ان ابو بكر الصديق ( رضي الله عنه) ابى الا ان يجاهد هؤلاء و يقضي على الفتنة. و ظهر ايضا من يقول اننا لا نريد الا ان نعيش بسلام وامان و نأمن على انفسنا واهلنا..فنجيب هؤلاء بتذكيرهم بقوله تعالى : (قُلْ إِن كَانَ آبَاؤُكُمْ وَأَبْنَآؤُكُمْ وَإِخْوَانُكُمْ وَأَزْوَاجُكُمْ وَعَشِيرَتُكُمْ وَأَمْوَالٌ اقْتَرَفْتُمُوهَا وَتِجَارَةٌ تَخْشَوْنَ كَسَادَهَا وَمَسَاكِنُ تَرْضَوْنَهَا أَحَبَّ إِلَيْكُم مِّنَ اللّهِ وَرَسُولِهِ وَجِهَادٍ فِي سَبِيلِهِ فَتَرَبَّصُواْ حَتَّى يَأْتِيَ اللّهُ بِأَمْرِهِ وَاللّهُ لاَ يَهْدِي الْقَوْمَ الْفَاسِقِينَ) [التوبة : 24] فهل بعد قول الله قول؟ قال تعالى: (وَمَا كَانَ لِمُؤْمِنٍ وَلَا مُؤْمِنَةٍ إِذَا قَضَى اللَّهُ وَرَسُولُهُ أَمْراً أَن يَكُونَ لَهُمُ الْخِيَرَةُ مِنْ أَمْرِهِمْ وَمَن يَعْصِ اللَّهَ وَرَسُولَهُ فَقَدْ ضَلَّ ضَلَالاً مُّبِيناً) [الأحزاب : 36] فحصل بعد ان زين الشيطان لبعض الناس ان الخير كل الخير بترك الجهاد ان ظهرت ما تسمى بالصحوات في الانبار فتحججوا برغبتهم باعادة الامن و الاستقرار الى مدينتهم و نسوا ان الامن والامان من عند الله وحده. و قامت هذه الصحوات بافعال شنيعة بمقاتلة كل من حمل السلاح ورفع رايه الجهاد و باتت اليد الضاربه للكافر الامريكي في هذه المدينة بعد ان عصيت عليه كل السبل الاخرى.. فوالت هذه الصحوات الامريكان ووضعت يدها بيد من جاء لحرب الاسلام واهله.. ثم لم ينته الامر عند ذلك بل ان ما تسمى بالصحوات قادت المدينة للاستسلام للمشروع الامريكي المتمثل بحكومة المنطقة الخضراء فجعلت الحكم ليس لله بل لحكومة اسست و شكلت وارسيت دعائمها من قبل المحتل الكافر ..حكومة تركت الاحتكام الى القران والسنة و اخذت تحتكم الى قوانين وضعيه يشرعها من لا يعرف لله قدرا....صار في عرفهم المسلم وغير المسلم سواء مادام هو عراقي وهذا والله مخالف لشرع الله.. فكيف يكون من يوحد الله ويعبده حق عبادته كمن خالف وعصى الله بل كمن كفر بالله وعبد الشيطان مثل ما يسمى باليزيدية الذين تضمن لهم هذه الحكومة حقوقهم كما هي لجميع المسلمين حسب قوانينهم.. فاعلموا هداكم الله ان الحكم لا يكون الا لله فقد قال تعالى : (وَقَاتِلُوهُمْ حَتَّى لاَ تَكُونَ فِتْنَةٌ وَيَكُونَ الدِّينُ لِلّهِ فَإِنِ انتَهَواْ فَلاَ عُدْوَانَ إِلاَّ عَلَى الظَّالِمِينَ) [البقرة : 193] و قال تعالى : (أَفَحُكْمَ الْجَاهِلِيَّةِ يَبْغُونَ وَمَنْ أَحْسَنُ مِنَ اللّهِ حُكْماً لِّقَوْمٍ يُوقِنُونَ) [المائدة : 50] وقال تعالى: (أَفَغَيْرَ اللّهِ أَبْتَغِي حَكَماً وَهُوَ الَّذِي أَنَزَلَ إِلَيْكُمُ الْكِتَابَ مُفَصَّلاً وَالَّذِينَ آتَيْنَاهُمُ الْكِتَابَ يَعْلَمُونَ أَنَّهُ مُنَزَّلٌ مِّن رَّبِّكَ بِالْحَقِّ فَلاَ تَكُونَنَّ مِنَ الْمُمْتَرِينَ )[الأنعام : 114] و قال تعالى: (وَكَذَلِكَ أَنزَلْنَاهُ حُكْماً عَرَبِيّاً وَلَئِنِ اتَّبَعْتَ أَهْوَاءهُم بَعْدَ مَا جَاءكَ مِنَ الْعِلْمِ مَا لَكَ مِنَ اللّهِ مِن وَلِيٍّ وَلاَ وَاقٍ) [الرعد : 37] وقال تعالى : (وَأَنزَلْنَا إِلَيْكَ الْكِتَابَ بِالْحَقِّ مُصَدِّقاً لِّمَا بَيْنَ يَدَيْهِ مِنَ الْكِتَابِ وَمُهَيْمِناً عَلَيْهِ فَاحْكُم بَيْنَهُم بِمَا أَنزَلَ اللّهُ وَلاَ تَتَّبِعْ أَهْوَاءهُمْ عَمَّا جَاءكَ مِنَ الْحَقِّ لِكُلٍّ جَعَلْنَا مِنكُمْ شِرْعَةً وَمِنْهَاجاً وَلَوْ شَاء اللّهُ لَجَعَلَكُمْ أُمَّةً وَاحِدَةً وَلَـكِن لِّيَبْلُوَكُمْ فِي مَا آتَاكُم فَاسْتَبِقُوا الخَيْرَاتِ إِلَى الله مَرْجِعُكُمْ جَمِيعاً فَيُنَبِّئُكُم بِمَا كُنتُمْ فِيهِ تَخْتَلِفُونَ )[المائدة : 48] و هذه كلها ادلة من كتاب الله على انه لا حكم الا حكم الله..وهذا بيان على بطلان حكومة المنطقة الخضراء. فاعلموا هداكم الله ان طريق الخلاص هو بالعودة الى الله تائبين قال تعالى: (قُلْ يَا عِبَادِيَ الَّذِينَ أَسْرَفُوا عَلَى أَنفُسِهِمْ لَا تَقْنَطُوا مِن رَّحْمَةِ اللَّهِ إِنَّ اللَّهَ يَغْفِرُ الذُّنُوبَ جَمِيعاً إِنَّهُ هُوَ الْغَفُورُ الرَّحِيمُ) [الزمر : 53] ثم الجهاد في سبيل الله الذي هو ذروة سنام الاسلام كما اخبر النبي صلى الله عليه وسلم. فيجب العودة الى كتاب الله وسنة نبيه والجهاد في سبيل الله حتى يقام شرع الله في الارض وحتى لا يكون حكم الا حكم الله سبحانه. سبحانك اللهم و بحمدك اشهد ان لا اله الا انت وحدك استغفرك واتوب اليك. الراجي عفو ربه ابو سليمان الناصر
