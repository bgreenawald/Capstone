سبَقــــها تواصلٌ اجْتماعي مع سكّـــــان الجَنوب على مدَى 6 سنوات (أي من بعــــد التحرير) في سياسةٍ اجْتمــــاعيّة ناجِحة ، بغضّ النظر عن أسباب نجـَـــاحها من إمكانيّات مــَــاديّة ، لكنّ الأهـــــمّ أن أجْواء الحرب كانت مهيّئــــة ، والنّسيج الاجتماعي مسْتــَـــعدٌ للتضحية ، فجاءَت الحرب بإمْكانيــّـــات عسكرية تعرف ما الذي يُمكنـُـــها فعله ، وسياســَـــة ردْعٍ ناجِحـــةْ (صواريخ الكاتْيُوشــــا) اتّضحت مِن أول يوم في الحـَـــرب ، مَا يهمّنـــــي في هذِه التجربة هو الوُقوف على مسألة (المِصداقيّـــــة) ، يقُول الجنرال جي.إف فولير رئِيس أركـــَـان قُوات التّاج البِريطاني : (إذا لَم يكـــن هناك تواصلٌ بْين الأحــــداث الباديَة للعَيـــان والأفْكار الّتــــي تُمثلها ، فسَوف يكــــون هناك معْــــركتان في هذه الحَالة : مَعركةٌ خياليـــّـة في مركز القــِــيادة ، ومعركة فــِــعلية على خطّ النار يكُون فيــــها العـَـــدو الحقِيقي موجـُـــوداً في مركَز قِيادتك .) ، فأيـْــــن المِصداقيّة التي تتمتّع بهـــــا القياداتُ الجِهادية حينَمــــا تُريد أن يصدّقها عَدوّهــــا ، أو حتّى أنصــــارُها ممّن هم مُغيّبون عن الحقِيقَـــــة !؟ وســَـــاحة العملِ لا تعرف إلا الأفــــعَالْ ، لا الأشرطــــة المرئية والصوتية والمقالات المكتـُــوبة ، إنّ المصداقيّة العسْكـــــرية الّتي كانت على لِسان زعيم الحـــِـزب حسن نصر الله ، أكْسَبــــت المعركة بُعداً آخــَــر ، تمثّل في إمْكَانيـــّــة تحقيقِ سياسة ردع بإمكانيّــــات ماديّة وعسكريّــــة ليست بالعاليـــَـة ، تُجبـــِــر الخَصم على وقْفِ خططــــه ، أو على الأقل تغييــــرها تجنباً لمصداقية وردع الخصــــم الآخر ، حتّـــــى لو كانَ الخصم أقوى رابع جيشٍ في العالَـــــم ، وهذا ما لمســــناه جدياً في قصف مدن فلسطيـــــن المحتلة تباعاً بناءً على التّهديدات ، ثُم التواصل الميْدانـــــي معَ المعركة ، فلم يخْجل حزب اللّــــه مِن أنْ يقدم اعتذاراً على لسـَـــان أكبر شخصيةٍ فيه بسبَب مقتــــلِ بعض الفلسطينييـــن في النّاصرة وبيسَــــان إثْر صواريخِه المُتساقطـــــة على مدِن الـ48 ، وقياداتُنـــــا الجِهادية ، وللّــــه الحمد ، لا تُبالي بعشــــَـرات المُسلمين الذين يسْقطـــــون في عمليّاتها ، حتى لوْ كـَـــان عددهم يفوق عدَد القتــْـــلى من العدُو ، والفتْوى طبـــــعاً عند الخِرفان جاهــِــزة .. (تتـــرس!) ، والحقّ كـَـــان على المقتُول ، مين اللي حكـــَـاله يكون هناك !؟ ثــُـــم تُختتم الديبـَــــاجة بقولِهم (يُبعثوا على نيّـــــاتِهم) ،! هذا عــَــدا عن سياسة التهلِيل والتّكبير والصيــــَـاح العجيب ولو حتّى كان إنفجار إطار دراجة هوَائية في مضِيق الدردنيل ! وليـْــس أدلّ من ذلك إلاّ التهلِيــــل لقتل عشراتٍ فــــي ديالَى والرّمــــادي قبل سنة ونيْف تقريبــــاً ، والتّفجيــــرُ الأوّل في شـــَـارعٍ عام قريب ٍمن الــــسّوق ووسْط دوَائـــِـر رسميّة مدنيّة يرتـــَـادها مُراجعون من عـــَـوام المُسلمين ، والآخر في مطعم شعبي ، وطَبــــعاً لا تسْأل عن التّبرِيــــرات التي تَدفعُهم لأن يدَافـــِـعُوا عن القَـتلة ، وأن المقْتوليـــن (علــَـى النّار مباشرة دُون إنتـِـــظار) ، وأن التّترس ، كما ذكـَـــرتُ ، يجيزُ لهُم ذلــــك ، أوْ كما علّق أحد الإخـــْـوَة ، وأضحَكني حقِيقــــةً ، حِينَما قال ( راح يُقولـــولك مُجتهد ، وهيك هيك كَسبــــان الأجر ، وكفى اللّــــه المؤمنين القتال! ) وحسبنا الله ونِعم الوكيل ، الطريــــف في الأمر أن دولة العـــــراق الإسلامية تبرّأت من الحادثيْــــن على لسان وزارة الإعـــــلام فيها ، التي تذكّـــــرت أن تضع بياناً بعد 5 أيامٍ من الحـــــادِثة !، لعلّـــــي أستبق العناويـــــن لأتطرّق لمسألةِ وزارة الإعـــــلام هذِه ، فعلى سبيــــلِ المثال ، تتذكـــّـر أنْ تنْشُر بيَاناً عن عمليـــّـة إسْتِشهادية في 14-3-2008م ، وتــــارِيخُ العملية هو 27-12-2007م ، أي أن الفاصل الزمنــــي بين موعِد العملية ، وتاريخ الإعــــلانِ عنها ما يُقــــارِب 3 أشهر ! فيا لِهذه الوَزارة الفــــذّة ، ولا أدري لِماذا لا تَتأخّر بيانات بقيّــــة الفَصائل (كجيش الراشدين وأنصار الإسلام والتي تنزل يومياً) ، وإن تأخّــــرت فعلى أقصى حد أسبوعاً أو أسبوعيْـــــن ، كَمــــا تتأخروا أنتم ؟! لعلّه صـــِـدق الأمريكَان وكثيرٍ مــــن المراقبيِــــن لساحتكم الإعـــــلاميّة ، بأن جهازكم الإعلامـــِـي تلقّى ضربةً أفقدته صــــَـوابه حتّى بات ينشُر عملياتِ إطــــْـلاق الهَاون والصّواريخ والّتِي كان يستحـــِـي (تنظيمُ) القاعِدة في بلاد الرافــــِـدين مِن نشْر عمليات مُصورة تُظهِره وهو يطلق هــــَـاون أو صَاروخ ! فإذْ بـ(الدولـــَـةِ) تنشُرها ، سبحــــان الله !، ولعلّ بعـــْـض هذِه البيانات تَحمل في فحوَاها الكذِب المتعمّــــد ، ولا أدري ما السّبب ! فقد بحثت فِــــي ذاتِ مرّة في كلّ أرشِيف الأخْبار العالميّة والصــــّـديقة والمُحادية عن إنفجار (ولَو حتّى عُبوة) في منطقةٍ قالــــُـوا أنّ هناك عمَلِية استِشـْــــهاديّة حدثت بها ، فلم أجد ! ، والعمليّــــة الاستشهاديّة أمرٌ لا يُمـــــكِن اخفائُه كانفجارِ عبُـــــوّةٍ مثلاً ، هذا عدا عن حقيقــــَـة خِلاف ما يدّعونه من كوْنِ وجــــُـود (مقتلةٍ عظـــِـيمَة) وأبْصَر شو فِي عمليــــّـاتِهم الإستشهَادية ، وقد شاهـــَـدتّ عملِيّتان صُورِتا بكامِيـــــرا مُراقبةٍ ، الأولى إسْتِشهادي يفجـــّـر نفْسه في نُقطةِ تفتيشٍ أمريكيـــّـة بها جُندي واحِد ! والأخرَى لنُقطة تفتيــــشٍ عرَاقية بِها جندي واحدٍ أيضـــــاً !، لَقد وجــَـــد سُكان الجنوبِ اللبنانــــيّ من يضْمن لــــَـهم خسائرهم ، ورأوْا في الحـَـــرب هدفاً يُمكن أن يتَحقــّــق (تحريرُ الأســْــرى) ، ودليلــُـــهم ما حَصل من تَبادلٍ للأسرى من قبـــْـل ، مع سابقـــَــة انتصارٍ لحزْب اللـّــه في عــَـــام 2000م ، فتوافَــــرت كلّ الشروطِ اللازمة لإقْناع الحاضـــِـنَة الشعبية أن سـَـــحق العدُو وهزِيمتـــَــه أمر محقق ، فلم تبـــَـالي تِلك المرأة الّتِي خرجتْ من تحـــــتِ الأنقاض علَى شاشــــةِ الجَزيرة ، وقد فقَدت إبنيـْـــها وزوجَها ، وقالت : (كُلـّـــه .. فِدا السيد حســــن) ، الحِكمـــــةُ ضالّة المؤمن ، ودِراسة الاسْتراتيجيّــــات العسْكَرية لا تتْبع التصنِيف المذْهبـِـــي أو العقَائدي أو الدّيني حتى ! ، بــَـــلِ الغاية هي الاسْتفادة وهذا هُو المطلــــوب ، كَلِمــــاتٌ في أذن البعض : حمداً للّـــــهِ على سَلامتك يا أبُو بـِـــلال (فجْرِ الخـــلافَة) ، وحريٌ بــــكَ أن تشكرَ من كــَـــان واسطةً لخُروجِك من السجْــــن بعد تدخّل أحدِ الإخْوة لــَـــدى قيادة المباحث في غــــزّة ، شَارِحــــاً لهم وضْعك وأنّك مُحتــــرفُ مهرجاناتٍ كلاَميـــّة دونما عمَلٍ ، مؤكّدا لهــــم أن (أحمَد شبـــَـات) الّذي هو أنْت مُجرّد واحــِــد صدرُه حامي ، ونـَـــازِل يخبص ، ورُفِــــع القلمُ عن المجْنُون حتى يفيق ! ولا تنْســَـــى استِعراض عضلاتِك بمعرّفك الجَديـــــد ، يا شِيـــــخ خشوا في (السّبطــَــانة) أحسنلكو ، يا نَحلة ورْشةِ البــُــراق الفنيّة ، أتحسبـِـــينَ أنّ هؤلاء الخرفان أهلٌ لكـُـــل ما طلبَه منهم أبو مُصعب السّــــوري في مقــــَالِه الذي ذكرتيه !؟ إِذا كنتي تظنّيــــن ذلِكَ ، فاسْمحي لـــِـي أن أعترف لكِ بأنّكِ (خروفة) مِثــْــلهم ، فهؤُلاء لا يَصلحون إلا لِثـــُـلاثية النهّي النبوِيّ (القِيــــل والقَال ، وكثْرة السـّـــؤال ، وإضَاعة المال) ، كَتِيبــــة الأبصر شُو الإعْلاميـــّــة ، تعت المنْشـُـــورات واللي مِش عارف شو تَاعــــهم ، بلاش اسْتِهبـــــال ، وحاولُوا تفهموا الناس أوّل شــــِـي أدبيَات الصــّــراع والمَخاطِر التي تتهدد ديِنـَــــهم ، قبْل ما تحكُولهم مصطَلحــــاتٍ من طرَاز (أمير المؤمنِيـــــن أبو عمر البغْدادي !) أضحكـَـــني أحدُ الإخوة حينما تسـَـــاءل : ماذا لو سـَـــأل أحد العامّة ممّــــن سيصِله المنْشور عن الإسْم الحقيقي لأبو عــُــمر هذا ! ، ورطـَــــة يا خرفان (: عـَـــودٌ على بدء معَ مــــواعظِ ورقائق العلاّمــــة الغريب الحائلي ، نصحتُك من قبــــْـل بالابتعادِ عن أسلُوب الاستعـــْـلاء والمواعِظ التي تسْتخدمه ، فلا تـَـــعدو كوْنك طِفلٌ صغير في ولايـــَــة الكويت الأمْريكيّــــة ، فابتَعد عن أسلُــــوب البِشارات وأنّ البشائــِــرَ تتوالى ، والرّسائل المشفّــــرة ، وما بيْن السّطــــور ، وكلّ الكذِب الذِي تحترفُ صنـــــاعتَه أنْت والخرفان أمثالُك ، ولوْ كنت كمَـــــا تقولُ وتزعم فأسْمِعني في الجَزيــــرة خبر استهداف جنود أمريكان في بـَــــلدك أيّها الخروف !، صَحيــــح ، كم وصَل عددُ مُعرّفـــَــاتِك !؟ حِسبــَـــوي ، خلك في دِراستك ، وخلْ عنّـــــك سالْفة مَدد وسَط أفريقيا ، أوّل مرة أعــْــرف إن الهِنْد في وسط أفريقيــــا ! رَحيــــق ، إسمُك يوحِي بأنّك مغفــــل ، لا أدري لمَاذا ! وَحدة من اثنَتيــْــن ، إما أنّك صاحبُ العصـــَـا ذاتَه بمعرّف آخر ، أو أنّك تُحاول أن تقلّده في حمـَـــاقته واسْتِطلاعاته ! ، غريبٌ أمــــْـركم ، حتّى في الحماقةِ ليس عنـــْـدكم إبداع ، قال إِيــــش ، إستطلاَع للرانديّيـــــن فقط ، ! أقْسِــــم لكُم يا إخْوة أن هذا عِنوان موضُوعٍ لـــَــه ، وبكامِل قـــُـواه العقلية يُطالب الرّاندييــــن بالتصْوِيت ، يَعني مطلُوب فقَط من (العمـــَـلاء) أن يدْخُلوه!، إِذَا كان عددُ الدّاخلِيــــن للموضوع كمـَـــا أراهُ الآن 460 ، فســــَـلامٌ على الفلّوجـــــةِ ! أنا أشـــُـكّ في قدراتِك العقليّـــــة بعدمَا طرحت الموْضُوع ، ولِتسجّــــل الفلوجة أول معاق قاعِــــدي فِي العــَـــالم !، مُعتـــَــز دغْمش ، أتمنى أن تكُون جـــِــلدة رأْسِك مش خميــــلَة !، لا زَالت كلِمــَـــات معتز تبعك أبو السلفيّــــة الجِهادية ترنّ فـــِــي أذنيّ حينَما كان يترحّم علَى يــــاسِر عرفات ، و سمِيـــــح المَدهُون !، بـَــــاغي الهُدى ، حرّر الضّفـــــة تاعتك من سُلطة عبّاس ، ومن اليهـُــــود ، قبل أن تكونَ مستعداً لاسْتِقبـــــال المهاجرين !، الأخ : طــــالبُ الدّعاء ، نَصيحتــــي لكَ أن تبْتعد عن المُهاترَات مع هؤلاءِ الخرفان ، فلم يمضِ على نِقاشك معهُم إلا صفْحتيــــن ، وإذ بالسيّد (الجــــزائري) وأخوهُ في الحمَاقة (تلميذ) يجعلانِــــك من عُملاء راند ! واللّــــه إن أكْثر من يخدِم راند هم هـــَـؤلاء وأشكالهم ، بِما يتعمّدون إظهاره من أسـْـــلوب حوارٍ أقلّ ما يُقال فيه أنّـــــه (مُنحطّ) ! تماماً مثــْــلهم ، ومثل تَربِيــــتهم ، ما زِلــْـــت أتذكّر مناقشاتِــــك معهُم في الحِسبة حـــَـول (الشّيـــــعةِ) ، ولم يشأ أيٌ منْهُم بأن ينْبـــِــس ببنت شفة ، سِوى ردود عارضيّــــة ، من بابِ الخجَل لا أكثـــــر ، من طرفِ (حنبـــــلي) ، دعكَ منهم ، فصدّقـــــنِي لن يردوا على ما ذَكرت من كلامٍ وتساؤُلات ، لأنّـــــهم بِبساطة حمقى وجُهـــــلاء بامتياز ! يحسب الواحِد منـْــــهم أنّه على شيْء ، وهو أقلّ من أن يكــُـــون بحجْمِ (البكتِيــــرْيا) في عالم الجهاد ، أيـّـــها الخرفان : اتّعظوا مما فعــَـــلهُ بكم (فارِسُ الجنوب) علَى الجزِيـــــرة توك ، والغرِيبُ أنّكم تحاولـــُــون تبريرَ فضِيحتكم هنــَــاك بفضائِح أكبر !، شرّ البليّــــــة ما يُضحك فعلاً ، إِلى جميـــــع شِلَل جلْجلــَـــت في غزّة : أتى تونِي بليـــــر ، وجُون كِيــــري ، ووُفود من الكونْجرِس والإتـــّــحاد الأوربي ، أتوْا إلى غزّة ، فأيـــْــن كُنتم !، مِن الآن فصـــَــاعداً سأسمّيكُم (جلْجَلُوتــُــو) ، على غِــــرَار المُسلْسل اليَابانــــي الكرْتُوني القِتَالي الشـــّــهير (نارُوتـُــــو) ، فأنْتُم بحــــقٍ تنظيماتٌ كرتُونيّـــــة !، ويستـــمر مسلسل الفضائح ،،
