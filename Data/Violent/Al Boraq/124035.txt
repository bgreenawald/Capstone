:097: بعــــد أن ضــاقـت الســبل بعمــلاء أيــران وحكومتـــها الصفويــة في العــراق ،،، والتضيــيق على ســوريــا التي تعـتـــبر وبإمتــــياز ذراع أيـــران الذي تضــرب بــه أهـــل الســنة في العـــراق ...... صــدرت الأوامــر لدولــة القــاعـــدة في العــراق مــن أســيادهـــا في قـــم وطهـــران لتبــــني عملـــيات يــوم الأربعـــاء الدامي في بغــــداد رغــم همجيتـــها وتعارضــها مــع أبســـط تعـــالــيم الاسلام ......... بــــل مــع كــل مــا هـــو انســـاني ........ والمصيـــبة إن تبــني العملـــية تأخـــر كــثـــيرا عــن المألوف ولــــم يصـــدر إلا بعـــد أن ظهــرت الأدلــة والاعــترافات التي تؤكد ضلوع أيـــران وعملائهـــا بالعملـــــية .... !!!!! وقــد اعـــترف بعــض العملاء أنفســهم ومــن مجلــس بغـــداد الـــتابع لحــزب الدعــوة العمــيل الذي إنشــق عــن حـــزب الدعـــوة الايــراني الأم نتيـــجة الخلافــات على الســرقــات والمغــانــم ،،، حيث عرضـوا صـــور إحــدى الشاحــنات وسط بغـــداد .... !!!! ورغــم ذلك إنــــبرت دولــة القــاعــدة ... دولــة الشــر والرذيــلة .... وبـــكل وقــاحــة ودون خجــل ... بــل دون أدنى ضمـــير الى تبني العملـــية ... وإلصــاقها بنفسها ... لإبعـــاد التهــمة عـــن ايران الشـر لأن وضعـــها الداخــلي لا يتحــمل أي هـــزة أو مشكلـــة خــارجية !! ومــــاذا تبنــــت ... لقــد تبنــت قـــتل وجــرح المـــئات من المدنــين الأبريـــــاء ... وتدعــي إنهــا لا تخــــوض بــدم الأبريــــاء ........ فهـــل هـــناك جـــرمــا أبشـــع وأفضـــع مــــن هـــذا الجـــــرم ..؟! ودلــيلا أوضــح مــن هــذا الدلــيل للإجــرام والعمــالة وإلحـاق الأذى وتشــويه سمعــة الاســـلام والمسلمــين وبأهــل الســنة خـــاصة .. ؟ ! فهــل هــناك منصــف وصــاحب ضمـــير بعــد كل هـــذا سيقــول إن دولة القــاعــدة في العـــراق ... بــل القــاعــدة بــكل فروعـــها ليــس لهــا صـــلة بايران وإنهـــا ليس ورقــة بيــد الدولة الصفــوية .. تلعـب بهــا أيــن ومـــتى تشـــاء ... بـــل هـــي إحـــدى الميليشيات العمــيلة المجــرمــة بإمتــــياز ... حــالهــا حــال جيش المهدي ومنظمة بـــدر والمجلس الأعلى وحــزب الدعـــوة .......... شـــلت أيـــدي العمـــلاء والمجــرمـين وفضحهـــم الله هــم وأسيادهـم في الدنـــــيا والأخـــــرة ......... يا ودود يا ودود ، ياذا العرش المجيد ، يا مبدئ يا معيد ، يا فعالا لما يريد ، أسألك بنور وجهك الذي ملىء أركان عرشك ، وأسألك بقدرتك التي قدرت بها على جميع خلقك و أسألك برحمتك التي و سعت كل شيء ، لا إله إلا أنت ... يا مغيث أغث العراق والعراقيـين يا مغيث أغث العراق والعراقيين يا مغيث أغث العراق والعراقيين اللهم صل وسلم على نبيك وخليلك محمد ما تعاقب الليل والنهار. منقول
