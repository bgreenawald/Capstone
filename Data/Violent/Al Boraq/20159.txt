بسم الله الرحمن الرحيم نصــــر من الله و فــتـــح قـريــــب ــــــــــــــــــــــــــــــــــــــــــــــــــ ــــــــــ ـــــــــــ البيان ـ 35 ــــــــــــــــــــ . حصاد الابطال في الفوج الثاني للتشكيل ، ضمن ملاحقتهم للاقذار الصفويين .. . قام المجاهدون الابطال باغتيال ثلاثة من أفراد جيش الدجال الصدر ، و ذلك في منطقة الغزالية / بغداد ، بتاريخ 21/9/2006 . . قام المجاهدون البواسل في سرية ثانية ضمن الفوج باغتيال أحد عناصر الكفر و الاجرام في فيلق غدر التابع للمجلس الاعلى للثورة الشيطانية ، و ذلك في حي الرافدين بتاريخ 24/9/2006 . . سرية أخرى من مجاهدينا النشامى ، قاموا بالهجوم على أحد الأوغاد المجرمين في فيلق غدر ، و اصطياده في منطقة حي الرافدين بتاريخ 25 /9/2006 . الله اكبر و العزة للاسلام و المسلمين ـــــــــــــــــــــــــــــــــــــــــــ قيادة لواء علي بن ابي طالب للعمليات الخاصة
